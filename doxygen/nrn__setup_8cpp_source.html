77 // file format defined in cooperation with nrncore/src/nrniv/nrnbbcore_write.cpp
78 // single integers are ascii one per line. arrays are binary int or double
79 // Note that regardless of the gid contents of a group, since all gids are
80 // globally unique, a filename convention which involves the first gid
81 // from the group is adequate. Also note that balance is carried out from a
82 // per group perspective and launching a process consists of specifying
83 // a list of group ids (first gid of the group) for each process.
84 //
85 // <firstgid>_1.dat
86 // n_presyn, n_netcon
87 // output_gids (npresyn) with -(type+1000*index) for those acell with no gid
88 // netcon_srcgid (nnetcon) -(type+1000*index) refers to acell with no gid
89 // -1 means the netcon has no source (not implemented)
90 // Note that the negative gids are only thread unique and not process unique.
91 // We create a thread specific hash table for the negative gids for each thread
92 // when <firstgid>_1.dat is read and then destroy it after <firstgid>_2.dat
93 // is finished using it. An earlier implementation which attempted to
94 // encode the thread number into the negative gid
95 // (i.e -ith - nth*(type +1000*index)) failed due to not large enough
96 // integer domain size.
97 // Note that for file transfer it is an error if a negative srcgid is
98 // not in the same thread as the target. This is because there it may
99 // not be the case that threads in a NEURON process end up on same process
100 // in CoreNEURON. NEURON will raise an error if this
101 // is the case. However, for direct memory transfer, it is allowed that
102 // a negative srcgid may be in a different thread than the target. So
103 // nrn2core_get_dat1 has a last arg netcon_negsrcgid_tid that specifies
104 // for the negative gids in netcon_srcgid (in that order) the source thread.
105 //
106 // <firstgid>_2.dat
107 // n_real_cell, n_output, n_real_output, nnode
108 // ndiam - 0 if no mechanism has dparam with diam semantics, or nnode
109 // nmech - includes artcell mechanisms
110 // for the nmech tml mechanisms
111 // type, nodecount
112 // nidata, nvdata, nweight
113 // v_parent_index (nnode)
114 // actual_a, b, area, v (nnode)
115 // diam - if ndiam > 0. Note that only valid diam is for those nodes with diam semantics mechanisms
116 // for the nmech tml mechanisms
117 // nodeindices (nodecount) but only if not an artificial cell
118 // data (nodecount*param_size)
119 // pdata (nodecount*dparam_size) but only if dparam_size > 0 on this side.
120 // output_vindex (n_presyn) >= 0 associated with voltages -(type+1000*index) for acell
121 // output_threshold (n_real_output)
122 // netcon_pnttype (nnetcon) <=0 if a NetCon does not have a target.
123 // netcon_pntindex (nnetcon)
124 // weights (nweight)
125 // delays (nnetcon)
126 // for the nmech tml mechanisms that have a nrn_bbcore_write method
127 // type
128 // icnt
129 // dcnt
130 // int array (number specified by the nodecount nrn_bbcore_write
131 // to be intepreted by this side's nrn_bbcore_read method)
132 // double array
133 // #VectorPlay_instances, for each of these instances
134 // 4 (VecPlayContinuousType)
135 // mtype
136 // index (from Memb_list.data)
137 // vecsize
138 // yvec
139 // tvec
140 //
141 // The critical issue requiring careful attention is that a coreneuron
142 // process reads many coreneuron thread files with a result that, although
143 // the conceptual
144 // total n_pre is the sum of all the n_presyn from each thread as is the
145 // total number of output_gid, the number of InputPreSyn instances must
146 // be computed here from a knowledge of all thread's netcon_srcgid after
147 // all thread's output_gids have been registered. We want to save the
148 // "individual allocation of many small objects" memory overhead by
149 // allocating a single InputPreSyn array for the entire process.
150 // For this reason cellgroup data are divided into two separate
151 // files with the first containing output_gids and netcon_srcgid which are
152 // stored in the nt.presyns array and nt.netcons array respectively

250 void determine_inputpresyn() {
251  // allocate the process wide InputPreSyn array
252  // all the output_gid have been registered and associated with PreSyn.
253  // now count the needed InputPreSyn by filling the netpar::gid2in map
254  gid2in.clear();
255 
256  // now have to fill the new table
257  // do not need to worry about negative gid overlap since only use
258  // it to search for PreSyn in this thread.
259 
260  std::vector<InputPreSyn*> inputpresyn_;
261 
262  for (int ith = 0; ith < nrn_nthread; ++ith) {
263  NrnThread& nt = nrn_threads[ith];
264  // associate gid with InputPreSyn and increase PreSyn and InputPreSyn count
265  nt.n_input_presyn = 0;
266  // if single thread or file transfer then definitely empty.
267  std::vector<int>& negsrcgid_tid = nrnthreads_netcon_negsrcgid_tid[ith];
268  size_t i_tid = 0;
269  for (int i = 0; i < nt.n_netcon; ++i) {
270  int gid = nrnthreads_netcon_srcgid[ith][i];
271  if (gid >= 0) {
272  /// If PreSyn or InputPreSyn is already in the map
273  auto gid2out_it = gid2out.find(gid);
274  if (gid2out_it != gid2out.end()) {
275  /// Increase PreSyn count
276  ++gid2out_it->second->nc_cnt_;
277  continue;
278  }
279  auto gid2in_it = gid2in.find(gid);
280  if (gid2in_it != gid2in.end()) {
281  /// Increase InputPreSyn count
282  ++gid2in_it->second->nc_cnt_;
283  continue;
284  }
285 
286  /// Create InputPreSyn and increase its count
287  InputPreSyn* psi = new InputPreSyn;
288  ++psi->nc_cnt_;
289  gid2in[gid] = psi;
290  inputpresyn_.push_back(psi);
291  ++nt.n_input_presyn;
292  } else {
293  int tid = nt.id;
294  if (!negsrcgid_tid.empty()) {
295  tid = negsrcgid_tid[i_tid++];
296  }
297  auto gid2out_it = neg_gid2out[tid].find(gid);
298  if (gid2out_it != neg_gid2out[tid].end()) {
299  /// Increase negative PreSyn count
300  ++gid2out_it->second->nc_cnt_;
301  }
302  }
303  }
304  }
305 
306  // now, we can opportunistically create the NetCon* pointer array
307  // to save some memory overhead for
308  // "large number of small array allocation" by
309  // counting the number of NetCons each PreSyn and InputPreSyn point to.
310  // Conceivably the nt.netcons could become a process global array
311  // in which case the NetCon* pointer array could become an integer index
312  // array. More speculatively, the index array could be eliminated itself
313  // if the process global NetCon array were ordered properly but that
314  // would interleave NetCon from different threads. Not a problem for
315  // serial threads but the reordering would propagate to nt.pntprocs
316  // if the NetCon data pointers are also replaced by integer indices.
317 
318  // First, allocate the pointer array.
319  int n_nc = 0;
320  for (int ith = 0; ith < nrn_nthread; ++ith) {
321  n_nc += nrn_threads[ith].n_netcon;
322  }
323  netcon_in_presyn_order_.resize(n_nc);
324  n_nc = 0;
325 
326  // fill the indices with the offset values and reset the nc_cnt_
327  // such that we use the nc_cnt_ in the following loop to assign the NetCon
328  // to the right place
329  // for PreSyn
330  int offset = 0;
331  for (int ith = 0; ith < nrn_nthread; ++ith) {
332  NrnThread& nt = nrn_threads[ith];
333  for (int i = 0; i < nt.n_presyn; ++i) {
334  PreSyn& ps = nt.presyns[i];
335  ps.nc_index_ = offset;
336  offset += ps.nc_cnt_;
337  ps.nc_cnt_ = 0;
338  }
339  }
340  // for InputPreSyn
341  for (auto psi: inputpresyn_) {
342  psi->nc_index_ = offset;
343  offset += psi->nc_cnt_;
344  psi->nc_cnt_ = 0;
345  }
346 
347  inputpresyn_.clear();
348 
349  // with gid to InputPreSyn and PreSyn maps we can setup the multisend
350  // target lists.
351  if (use_multisend_) {
352 #if NRN_MULTISEND
353  nrn_multisend_setup();
354 #endif
355  }
356 
357  // fill the netcon_in_presyn_order and recompute nc_cnt_
358  // note that not all netcon_in_presyn will be filled if there are netcon
359  // with no presyn (ie. nrnthreads_netcon_srcgid[nt.id][i] = -1) but that is ok since they are
360  // only used via ps.nc_index_ and ps.nc_cnt_;
361  for (int ith = 0; ith < nrn_nthread; ++ith) {
362  NrnThread& nt = nrn_threads[ith];
363  // if single thread or file transfer then definitely empty.
364  std::vector<int>& negsrcgid_tid = nrnthreads_netcon_negsrcgid_tid[ith];
365  size_t i_tid = 0;
366  for (int i = 0; i < nt.n_netcon; ++i) {
367  NetCon* nc = nt.netcons + i;
368  int gid = nrnthreads_netcon_srcgid[ith][i];
369  int tid = ith;
370  if (!negsrcgid_tid.empty() && gid < -1) {
371  tid = negsrcgid_tid[i_tid++];
372  }
373  PreSyn* ps;
374  InputPreSyn* psi;
375  netpar_tid_gid2ps(tid, gid, &ps, &psi);
376  if (ps) {
377  netcon_in_presyn_order_[ps->nc_index_ + ps->nc_cnt_] = nc;
378  ++ps->nc_cnt_;
379  ++n_nc;
380  } else if (psi) {
381  netcon_in_presyn_order_[psi->nc_index_ + psi->nc_cnt_] = nc;
382  ++psi->nc_cnt_;
383  ++n_nc;
384  }
385  }
386  }
387 
388  /// Resize the vector to its actual size of the netcons put in it
389  netcon_in_presyn_order_.resize(n_nc);
390 }
