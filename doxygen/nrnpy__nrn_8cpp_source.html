1533 static PyObject* seg_volume(NPySegObj* self) {
1534  Section* sec = self->pysec_->sec_;
1535  CHECK_SEC_INVALID(sec);
1536  int i;
1537  if (sec->recalc_area_) {
1538  nrn_area_ri(sec);
1539  }
1540  double x = self->x_;
1541  double a = 0.0;
1542  // volume only exists on the interior
1543  if (x > 0. && x < 1.) {
1544  // every section owns one extra node (the conservation node)
1545  int nseg = sec->nnode - 1;
1546  double length = section_length(sec) / nseg;
1547  int iseg = x * nseg;
1548  double seg_left_arc = iseg * length;
1549  double seg_right_arc = (iseg + 1) * length;
1550  if (sec->npt3d > 1) {
1551  int i_left = arg_bisect_arc3d(sec, sec->npt3d, seg_left_arc);
1552  double left_diam = fabs(sec->pt3d[i_left].d);
1553  double right_diam = fabs(sec->pt3d[i_left + 1].d);
1554  double left_arc = seg_left_arc;
1555  left_diam = interpolate(
1556  sec->pt3d[i_left].arc, sec->pt3d[i_left + 1].arc, left_diam, right_diam, left_arc);
1557 
1558  for (i = i_left + 1; i < sec->npt3d && sec->pt3d[i].arc < seg_right_arc; i++) {
1559  right_diam = fabs(sec->pt3d[i].d);
1560  a += scaled_frustum_volume(sec->pt3d[i].arc - left_arc, left_diam, right_diam);
1561  left_arc = sec->pt3d[i].arc;
1562  left_diam = right_diam;
1563  }
1564  if (i < sec->npt3d) {
1565  right_diam = interpolate(
1566  left_arc, sec->pt3d[i].arc, left_diam, fabs(sec->pt3d[i].d), seg_right_arc);
1567  a += scaled_frustum_volume(seg_right_arc - left_arc, left_diam, right_diam);
1568  }
1569 
1570  // the scaled_frustum_volume formula factored out a pi/12
1571  a *= M_PI / 12;
1572  } else {
1573  // 0 or 1 3D points... so give cylinder volume
1574  Node* nd = node_exact(sec, x);
1575  for (Prop* p = nd->prop; p; p = p->next) {
1576  if (p->_type == MORPHOLOGY) {
1577  double diam = p->param(0);
1578  a = M_PI * diam * diam / 4 * length;
1579  break;
1580  }
1581  }
1582  }
1583  }
1584  PyObject* result = Py_BuildValue("d", a);
1585  return result;
1586 }
