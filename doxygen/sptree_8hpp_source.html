109  *
110  * spenq() -- insert item in a tree.
111  *
112  * put n in q after all other nodes with the same key; when this is
113  * done, n will be the root of the splay tree representing q, all nodes
114  * in q with keys less than or equal to that of n will be in the
115  * left subtree, all with greater keys will be in the right subtree;
116  * the tree is split into these subtrees from the top down, with rotations
117  * performed along the way to shorten the left branch of the right subtree
118  * and the right branch of the left subtree
119  */
120 template <typename SPBLK>
121 SPBLK* spenq(SPBLK* n, SPTREE<SPBLK>* q) {
122  SPBLK* left; /* the rightmost node in the left tree */
123  SPBLK* right; /* the leftmost node in the right tree */
124  SPBLK* next; /* the root of the unsplit part */
125  SPBLK* temp;
126 
127  double key;
128  int Sct; /* Strcmp value */
129 
130  n->uplink = nullptr;
131  next = q->root;
132  q->root = n;
133  if (next == nullptr) /* trivial enq */
134  {
135  n->leftlink = nullptr;
136  n->rightlink = nullptr;
137  } else /* difficult enq */
138  {
139  key = n->key;
140  left = n;
141  right = n;
142 
143  /* n's left and right children will hold the right and left
144  splayed trees resulting from splitting on n->key;
145  note that the children will be reversed! */
146 
147  q->enqcmps++;
148  if (STRCMP(next->key, key) > 0)
149  goto two;
150 
151  one: /* assert next->key <= key */
152 
153  do /* walk to the right in the left tree */
154  {
155  temp = next->rightlink;
156  if (temp == nullptr) {
157  left->rightlink = next;
158  next->uplink = left;
159  right->leftlink = nullptr;
160  goto done; /* job done, entire tree split */
161  }
162 
163  q->enqcmps++;
164  if (STRCMP(temp->key, key) > 0) {
165  left->rightlink = next;
166  next->uplink = left;
167  left = next;
168  next = temp;
169  goto two; /* change sides */
170  }
171 
172  next->rightlink = temp->leftlink;
173  if (temp->leftlink != nullptr)
174  temp->leftlink->uplink = next;
175  left->rightlink = temp;
176  temp->uplink = left;
177  temp->leftlink = next;
178  next->uplink = temp;
179  left = temp;
180  next = temp->rightlink;
181  if (next == nullptr) {
182  right->leftlink = nullptr;
183  goto done; /* job done, entire tree split */
184  }
185 
186  q->enqcmps++;
187 
188  } while (STRCMP(next->key, key) <= 0); /* change sides */
189 
190  two: /* assert next->key > key */
191 
192  do /* walk to the left in the right tree */
193  {
194  temp = next->leftlink;
195  if (temp == nullptr) {
196  right->leftlink = next;
197  next->uplink = right;
198  left->rightlink = nullptr;
199  goto done; /* job done, entire tree split */
200  }
201 
202  q->enqcmps++;
203  if (STRCMP(temp->key, key) <= 0) {
204  right->leftlink = next;
205  next->uplink = right;
206  right = next;
207  next = temp;
208  goto one; /* change sides */
209  }
210  next->leftlink = temp->rightlink;
211  if (temp->rightlink != nullptr)
212  temp->rightlink->uplink = next;
213  right->leftlink = temp;
214  temp->uplink = right;
215  temp->rightlink = next;
216  next->uplink = temp;
217  right = temp;
218  next = temp->leftlink;
219  if (next == nullptr) {
220  left->rightlink = nullptr;
221  goto done; /* job done, entire tree split */
222  }
223 
224  q->enqcmps++;
225 
226  } while (STRCMP(next->key, key) > 0); /* change sides */
227 
228  goto one;
229 
230  done: /* split is done, branches of n need reversal */
231 
232  temp = n->leftlink;
233  n->leftlink = n->rightlink;
234  n->rightlink = temp;
235  }
236 
237  return (n);
238

243  *
244  * spdeq() -- return and remove head node from a subtree.
245  *
246  * remove and return the head node from the node set; this deletes
247  * (and returns) the leftmost node from q, replacing it with its right
248  * subtree (if there is one); on the way to the leftmost node, rotations
249  * are performed to shorten the left branch of the tree
250  */
251 template <typename SPBLK>
252 SPBLK* spdeq(SPBLK** np) /* pointer to a node pointer */
253 
254 {
255  SPBLK* deq; /* one to return */
256  SPBLK* next; /* the next thing to deal with */
257  SPBLK* left; /* the left child of next */
258  SPBLK* farleft; /* the left child of left */
259  SPBLK* farfarleft; /* the left child of farleft */
260 
261  if (np == nullptr || *np == nullptr) {
262  deq = nullptr;
263  } else {
264  next = *np;
265  left = next->leftlink;
266  if (left == nullptr) {
267  deq = next;
268  *np = next->rightlink;
269 
270  if (*np != nullptr)
271  (*np)->uplink = nullptr;
272 
273  } else
274  for (;;) /* left is not null */
275  {
276  /* next is not it, left is not nullptr, might be it */
277  farleft = left->leftlink;
278  if (farleft == nullptr) {
279  deq = left;
280  next->leftlink = left->rightlink;
281  if (left->rightlink != nullptr)
282  left->rightlink->uplink = next;
283  break;
284  }
285 
286  /* next, left are not it, farleft is not nullptr, might be it */
287  farfarleft = farleft->leftlink;
288  if (farfarleft == nullptr) {
289  deq = farleft;
290  left->leftlink = farleft->rightlink;
291  if (farleft->rightlink != nullptr)
292  farleft->rightlink->uplink = left;
293  break;
294  }
295 
296  /* next, left, farleft are not it, rotate */
297  next->leftlink = farleft;
298  farleft->uplink = next;
299  left->leftlink = farleft->rightlink;
300  if (farleft->rightlink != nullptr)
301  farleft->rightlink->uplink = left;
302  farleft->rightlink = left;
303  left->uplink = farleft;
304  next = farleft;

306  }

315  *
316  * splay() -- reorganize the tree.
317  *
318  * the tree is reorganized so that n is the root of the
319  * splay tree representing q; results are unpredictable if n is not
320  * in q to start with; q is split from n up to the old root, with all
321  * nodes to the left of n ending up in the left subtree, and all nodes
322  * to the right of n ending up in the right subtree; the left branch of
323  * the right subtree and the right branch of the left subtree are
324  * shortened in the process
325  *
326  * this code assumes that n is not nullptr and is in q; it can sometimes
327  * detect n not in q and complain
328  */
329 
330 template <typename SPBLK>
331 void splay(SPBLK* n, SPTREE<SPBLK>* q) {
332  SPBLK* up; /* points to the node being dealt with */
333  SPBLK* prev; /* a descendent of up, already dealt with */
334  SPBLK* upup; /* the parent of up */
335  SPBLK* upupup; /* the grandparent of up */
336  SPBLK* left; /* the top of left subtree being built */
337  SPBLK* right; /* the top of right subtree being built */
338 
339 
340  left = n->leftlink;
341  right = n->rightlink;
342  prev = n;
343  up = prev->uplink;
344 
345  while (up != nullptr) {
346  /* walk up the tree towards the root, splaying all to the left of
347  n into the left subtree, all to right into the right subtree */
348 
349  upup = up->uplink;
350  if (up->leftlink == prev) /* up is to the right of n */
351  {
352  if (upup != nullptr && upup->leftlink == up) /* rotate */
353  {
354  upupup = upup->uplink;
355  upup->leftlink = up->rightlink;
356  if (upup->leftlink != nullptr)
357  upup->leftlink->uplink = upup;
358  up->rightlink = upup;
359  upup->uplink = up;
360  if (upupup == nullptr)
361  q->root = up;
362  else if (upupup->leftlink == upup)
363  upupup->leftlink = up;
364  else
365  upupup->rightlink = up;
366  up->uplink = upupup;
367  upup = upupup;
368  }
369  up->leftlink = right;
370  if (right != nullptr)
371  right->uplink = up;
372  right = up;
373 
374  } else /* up is to the left of n */
375  {
376  if (upup != nullptr && upup->rightlink == up) /* rotate */
377  {
378  upupup = upup->uplink;
379  upup->rightlink = up->leftlink;
380  if (upup->rightlink != nullptr)
381  upup->rightlink->uplink = upup;
382  up->leftlink = upup;
383  upup->uplink = up;
384  if (upupup == nullptr)
385  q->root = up;
386  else if (upupup->rightlink == upup)
387  upupup->rightlink = up;
388  else
389  upupup->leftlink = up;
390  up->uplink = upupup;
391  upup = upupup;
392  }
393  up->rightlink = left;
394  if (left != nullptr)
395  left->uplink = up;
396  left = up;
397  }
398  prev = up;
399  up = upup;
400  }
401 
402 #ifdef DEBUG
403  if (q->root != prev) {
404  /* fprintf(stderr, " *** bug in splay: n not in q *** " ); */
405  abort();
406  }
407 #endif
408 
409  n->leftlink = left;
410  n->rightlink = right;
411  if (left != nullptr)
412  left->uplink = n;
413  if (right != nullptr)
414  right->uplink = n;
415  q->root = n;
416  n->uplink = nullptr;
417 
418 } /* splay */

492 /*----------------
493  *
494  * sphead() -- return the "lowest" element in the tree.
495  *
496  * returns a reference to the head event in the event-set q,
497  * represented as a splay tree; q->root ends up pointing to the head
498  * event, and the old left branch of q is shortened, as if q had
499  * been splayed about the head element; this is done by dequeueing
500  * the head and then making the resulting queue the right son of
501  * the head returned by spdeq; an alternative is provided which
502  * avoids splaying but just searches for and returns a pointer to
503  * the bottom of the left branch
504  */
505 template <typename SPBLK>
506 SPBLK* sphead(SPTREE<SPBLK>* q) {
507  SPBLK* x;
508 
509  /* splay version, good amortized bound */
510  x = spdeq(&q->root);
511  if (x != nullptr) {
512  x->rightlink = q->root;
513  x->leftlink = nullptr;
514  x->uplink = nullptr;
515  if (q->root != nullptr)
516  q->root->uplink = x;
517  }
518  q->root = x;
519 
520  /* alternative version, bad amortized bound,
521  but faster on the average */
522 
523  return (x);
524 
525 } /* sphead */
