11 /**
12  * @brief Non-template stable handle to a generic value.
13  *
14  * This is a type-erased version of data_handle<T>, with the additional feature
15  * that it can store values of POD types no larger than a pointer (typically int
16  * and float). It stores (at runtime) the type of the value it contains, and is
17  * therefore type-safe, but this increases sizeof(generic_data_handle) by 50% so
18  * it may be prudent to view this as useful for validation/debugging but not
19  * something to become too dependent on.
20  *
21  * There are several states that instances of this class can be in:
22  * - null, no value is contained, any type can be assigned without any type
23  * mismatch error (m_type=null, m_container=null, m_offset=null)
24  * - wrapping an instance of a small, trivial type T (m_type=&typeid(T),
25  * m_container=the_value, m_offset=null)
26  * - wrapping a data handle<T> (m_type=&typeid(T*), m_container=the_container,
27  * m_offset=ptr_to_row)
28  *
29  * @todo Consider whether this should be made more like std::any (with a maximum
30  * 2*sizeof(void*) and a promise never to allocate memory dynamically) so
31  * it actually has a data_handle<T> subobject. Presumably that would mean
32  * data_handle<T> would need to have a trivial destructor. This might make
33  * it harder in future to have some vector_of_generic_data_handle type
34  * that hoists out the pointer-to-container and typeid parts that should
35  * be the same for all rows.
36  */

119  /**
120  * @brief Create data_handle<T> from a generic data handle.
121  *
122  * The conversion will succeed, yielding a null data_handle<T>, if the
123  * generic data handle is null. If the generic data handle is not null then
124  * the conversion will succeed if the generic data handle actually holds a
125  * data_handle<T> or a literal T*.
126  *
127  * It might be interesting in future to explore dropping m_type in optimised
128  * builds, in which case we should aim to avoid predicating important logic
129  * on exceptions thrown by this function.
130  */
131  template <typename T>
132  explicit operator data_handle<T>() const {
133  // Either the type has to match or the generic_data_handle needs to have
134  // never been given a type
135  if (!m_type) {
136  // A (typeless / default-constructed) null generic_data_handle can
137  // be converted to any (null) data_handle<T>.
138  return {};
139  }
140  if (typeid(T*) != *m_type) {
141  throw_error(" cannot be converted to data_handle<" + cxx_demangle(typeid(T).name()) +
142  ">");
143  }
144  if (m_offset.has_always_been_null()) {
145  // This is a data handle in backwards-compatibility mode, wrapping a
146  // raw pointer of type T*, or a null handle that has always been null (as opposed to a
147  // handle that became null). Passing do_not_search prevents the data_handle<T>
148  // constructor from trying to find the raw pointer in the NEURON data structures.
149  return {do_not_search, static_cast<T*>(m_container)};
150  }
151  // Nothing after here is reachable with T=void, as data_handle<void> is always either null
152  // or in backwards-compatibility mode. the branch structure has been chosen to try and
153  // minimise compiler warnings and maximise reported coverage...
154  if constexpr (!std::is_same_v<T, void>) {
155  if (!m_offset.was_once_valid()) {
156  // A real and still-valid data handle. This cannot be instantiated with T=void
157  // because data_handle<void> does not have a 4-argument constructor.
158  assert(m_container);
159  return {m_offset, static_cast<T* const*>(m_container), m_array_dim, m_array_index};
160  }
161  }
162  // Reaching here should mean T != void && was_once_valid == true, i.e. this used to be a
163  // valid data handle, but it has since been invalidated. Invalid data handles never become
164  // valid again, so we can safely produce a "fully null" data_handle<T>.
165  return {};
166  }

168  /** @brief Explicit conversion to any T.
169  *
170  * It might be interesting in future to explore dropping m_type in
171  * optimised builds, in which case we should aim to avoid predicating
172  * important logic on exceptions thrown by this function.
173  *
174  * Something like static_cast<double*>(generic_handle) will work both if
175  * the Datum holds a literal double* and if it holds a data_handle<double>.
176  *
177  * @todo Consider conversion to bool and whether this means not-null or to
178  * obtain a literal, wrapped bool value
179  */
180  template <typename T>
181  T get() const {
182  if constexpr (std::is_pointer_v<T>) {
183  // If T=U* (i.e. T is a pointer type) then we might be in modern
184  // mode, go via data_handle<U>
185  return static_cast<T>(static_cast<data_handle<std::remove_pointer_t<T>>>(*this));
186  } else {
187  // Getting a literal value saved in m_container
188  static_assert(can_be_stored_literally_v<T>,
189  "generic_data_handle can only hold non-pointer types that are trivial "
190  "and smaller than a pointer");
191  if (!m_offset.has_always_been_null()) {
192  throw_error(" conversion to " + cxx_demangle(typeid(T).name()) +
193  " not possible for a handle [that was] in modern mode");
194  }
195  if (typeid(T) != *m_type) {
196  throw_error(" does not hold a literal value of type " +
197  cxx_demangle(typeid(T).name()));
198  }
199  T ret{};
200  std::memcpy(&ret, &m_container, sizeof(T));
201  return ret;
202  }

248  /** @brief Obtain a reference to the literal value held by this handle.
249  *
250  * Storing literal values is incompatible with storing data_handle<T>. If
251  * the handle stores data_handle<T> then calling this method throws an
252  * exception. If the handle is null, this sets the stored type to be T and
253  * returns a reference to it. If the handle already holds a literal value
254  * of type T then a reference to it is returned.
255  *
256  * Note that, unlike converting to double*, literal_value<double*>() will
257  * fail if the handle contains data_handle<double>, as in that case there
258  * is no persistent double* that could be referred to.
259  *
260  * It might be interesting in future to explore dropping m_type in
261  * optimised builds, in which case we should aim to avoid predicating
262  * important logic on exceptions thrown by this function.
263  */
264  template <typename T>
265  [[nodiscard]] T& literal_value() {
266  if (!m_offset.has_always_been_null()) {
267  throw_error("::literal_value<" + cxx_demangle(typeid(T).name()) +
268  "> cannot be called on a handle [that was] in modern mode");
269  } else {
270  // This is a data handle in backwards-compatibility mode, wrapping a
271  // raw pointer, or a null data handle. Using raw_ptr() on a typeless_null
272  // (default-constructed) handle turns it into a legacy handle-to-T.
273  if (!m_type) {
274  m_type = &typeid(T);
275  } else if (typeid(T) != *m_type) {
276  throw_error(" does not hold a literal value of type " +
277  cxx_demangle(typeid(T).name()));
278  }
279  return *reinterpret_cast<T*>(&m_container);
280  }
281  }
