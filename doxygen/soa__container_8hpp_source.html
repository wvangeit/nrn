924  /**
925  * @brief Create a token guaranteeing the container is in "frozen" state.
926  *
927  * This does *not* modify the "sorted" flag on the container.
928  *
929  * The token type is copy constructible but not default constructible.
930  * There is no need to check if a given instance of the token type is
931  * "valid"; if a token is held then the container is guaranteed to be
932  * frozen.
933  *
934  * The tokens returned by this function are reference counted; the
935  * container will be frozen for as long as any token is alive.
936  *
937  * Methods such as apply_reverse_permutation() take a non-const reference
938  * to one of these tokens. This is because a non-const token referring to a
939  * container with a token reference count of exactly one has an elevated
940  * status: the holder can lend it out to methods such as
941  * apply_reverse_permutation() to authorize specific pointer-invaliding
942  * operations. This is useful for implementing methods such as
943  * nrn_ensure_model_data_are_sorted() in a thread-safe way.
944  *
945  * This method can be called from multiple threads, but note that doing so
946  * can have surprising effects w.r.t. the elevated status mentioned in the
947  * previous paragraph.
948  *
949  * It is user-defined precisely what "sorted" means, but the soa<...> class
950  * makes some guarantees:
951  * - if the container is frozen, no pointers to elements in the underlying
952  * storage will be invalidated -- attempts to do so will throw or abort.
953  * - if the container is not frozen, it will remain flagged as sorted until
954  * a potentially-pointer-invalidating operation (insertion, deletion)
955  * occurs, or mark_as_unsorted() is called.
956  * To mark a container as "sorted", apply an explicit permutation to it.
957  *
958  * Note that "frozen" refers to the storage layout, not to the stored value,
959  * meaning that values inside a frozen container can still be modified --
960  * "frozen" is not "runtime const".
961  *
962  * @todo A future extension could be to preserve the sorted flag until
963  * pointers are actually, not potentially, invalidated.
964  */

967  std::lock_guard _{m_mut};

969  ++m_frozen_count;

1050  /**
1051  * @brief Permute the SoA-format data using an arbitrary range of integers.
1052  * @param permutation The reverse permutation vector to apply.
1053  * @param token A non-const token demonstrating that the caller is the only
1054  * party that is forcing the container to be frozen, and (non-const)
1055  * that they are authorised to transfer that status into this method
1056  */
1057  template <typename Range>
1058  void apply_reverse_permutation(Range permutation, frozen_token_type& sorted_token) {
1059  // Check that the given vector is a valid permutation of length size().
1060  // The existence of `sorted_token` means that my_size cannot become
1061  // invalid, even though we don't hold `m_mut` yet.
1062  std::size_t const my_size{size()};
1063  bool const is_trivial{detail::check_permutation_vector(permutation, my_size)};
1064  // Lock access to m_frozen_count and m_sorted.
1065  std::lock_guard _{m_mut};
1066  // Applying a permutation in general invalidates indices, so it is
1067  // forbidden if the structure is frozen, and it leaves the structure
1068  // unsorted. We therefore require that the frozen count is 1, which
1069  // corresponds to the `sorted_token` argument to this function being
1070  // the only active token.
1071  if (m_frozen_count != 1) {
1072  throw_error(
1073  "apply_reverse_permutation() given a token that was not the only valid one");
1074  }
1075  if (!is_trivial) {
1076  // Now we apply the reverse permutation in `permutation` to all of the columns in the
1077  // container. This is the algorithm from boost::algorithm::apply_reverse_permutation.
1078  for (std::size_t i = 0; i < my_size; ++i) {
1079  while (i != permutation[i]) {
1080  using ::std::swap;
1081  auto const next = permutation[i];
1082  for_each_vector<detail::may_cause_reallocation::No>(
1083  [i, next](auto const& tag, auto& vec, auto field_index, auto array_dim) {
1084  // swap the i-th and next-th array_dim-sized sub-ranges of vec
1085  ::std::swap_ranges(::std::next(vec.begin(), i * array_dim),
1086  ::std::next(vec.begin(), (i + 1) * array_dim),
1087  ::std::next(vec.begin(), next * array_dim));
1088  });
1089  swap(permutation[i], permutation[next]);
1090  }
1091  }
1092  // update the indices in the container
1093  for (auto i = 0ul; i < my_size; ++i) {
1094  m_indices[i].set_current_row(i);
1095  }
1096  // If the container was previously marked sorted, and we have just
1097  // applied a non-trivial permutation to it, then we need to call the
1098  // callback if it exists (to invalidate any caches based on the old
1099  // sort order).
1100  if (m_sorted && m_unsorted_callback) {
1101  m_unsorted_callback();
1102  }
1103  }
1104  // In any case, applying a permutation leaves the container in sorted
1105  // state. The caller made an explicit choice about which element should
1106  // live where, which is as much as we can hope for.
1107  m_sorted = true;
1108  }

1139  * unique_ptr, decreases size() by one.
1140  *
1141  * Note that this has different semantics to standard library container
1142  * methods such as emplace_back(), push_back(), insert() and so on. Because
1143  * the returned identifier manages the lifetime of the newly-created entry,
1144  * discarding the return value will cause the new entry to immediately be
1145  * deleted.
1146  *
1147  * This is a low-level call that is useful for the implementation of the
1148  * owning_identifier template. The returned owning identifier is typically

1150  * methods (e.g. v(), v_handle() for a Node).

1152  [[nodiscard]] owning_identifier<Storage> acquire_owning_identifier() {
1153  // Lock access to m_frozen_count and m_sorted.
1154  std::lock_guard _{m_mut};
1155  if (m_frozen_count) {
1156  throw_error("acquire_owning_identifier() called on a frozen structure");

1159  // reallocation and, therefore, invalidation of pointers. At present,
1160  // "sorted" is defined to mean that pointers have not been invalidated.
1161  // There are two reasonable changes that could be made here:
1162  // - possibly for release builds, we could only mark unsorted if a
1163  // reallocation *actually* happens
1164  // - "sorted" could be defined to mean that indices have not been
1165  // invalidated -- adding a new entry to the end of the container
1166  // never invalidates indices
1167  mark_as_unsorted_impl<true>();
