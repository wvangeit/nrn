281 void NrnDAE::dkres(double* y, double* yprime, double* delta) {
282  // printf("NrnDAE::dkres %lx\n", (long)this);
283  // delta is already f(y)
284  // now subtract c*y'
285  // The problem is the map between y and yprime and the local

290  yptmp_[i] = yprime[bmap_[i] - 1];
291  }

297  // the identity that it will stay that way

301  cyp = &cyp_;
