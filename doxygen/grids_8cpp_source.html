479 void ECS_Grid_node::set_tortuosity(PyHocObject* my_permeability) {
480  /*Check to see if variable tortuosity/volume fraction is used*/
481  /*note permeability is the 1/tortuosity^2*/
482  if (PyFloat_Check(my_permeability)) {
483  if (get_permeability == &get_permeability_scalar) {
484  double new_permeability = PyFloat_AsDouble((PyObject*) my_permeability);
485  get_permeability = &get_permeability_scalar;
486  dc_x *= new_permeability / permeability[0];
487  dc_y *= new_permeability / permeability[0];
488  dc_z *= new_permeability / permeability[0];
489  permeability[0] = new_permeability;
490  } else {
491  permeability = (double*) malloc(sizeof(double));
492  permeability[0] = PyFloat_AsDouble((PyObject*) my_permeability);
493  /* don't free the old permeability -- let python do it */
494  dc_x *= permeability[0];
495  dc_y *= permeability[0];
496  dc_z *= permeability[0];
497  get_permeability = &get_permeability_scalar;
498  VARIABLE_ECS_VOLUME = (VARIABLE_ECS_VOLUME == TORTUOSITY) ? FALSE : VARIABLE_ECS_VOLUME;
499  }
500  } else {
501  if (get_permeability == &get_permeability_scalar) {
502  /* rescale to remove old permeability*/
503  dc_x /= permeability[0];
504  dc_y /= permeability[0];
505  dc_z /= permeability[0];
506  free(permeability);
507  permeability = static_cast<double*>(my_permeability->u.px_);
508  VARIABLE_ECS_VOLUME = (VARIABLE_ECS_VOLUME == FALSE) ? TORTUOSITY : VARIABLE_ECS_VOLUME;
509  get_permeability = &get_permeability_array;
510  } else {
511  permeability = static_cast<double*>(my_permeability->u.px_);
512  }
513  }
