272  convert = 1;
273  islong = 0;
274  sawnum = 0;
275  ++pf;
276  if (!*pf)
277  goto incomplete;
278  if (*pf == '*') {
279  convert = 0;

281  if (!*pf)
282  goto incomplete;
283  }

286  }

288  sawnum = 1;
289  ++pf;
290  if (!*pf)
291  goto incomplete;
292  }

294  islong = 1;
295  ++pf;
296  if (!*pf)
297  goto incomplete;
298  }
299  if (convert)
300  switch (*pf) {
301  case '%':
302  convert = 0;
303  break;
304  case 'd':
305  case 'i':
306  case 'o':
307  case 'u':
308  case 'x':

312  } else {

315  }
316  break;
317  case 'e':
318  case 'f':

323  } else {

326  }
327  break;

331  }

333  for (;;) {
334  if (!*pf)
335  goto incomplete;
336  if (*pf == ']') {
337  if (!((i == 1) || ((i == 2) && (pf[-1] == '^')))) {
338  break;
339  }

342  ++pf;
343  }
344  case 's':
345  if (islong) {
346  goto bad_specifier;
347  }
348  arg[iarg].type = 's';
349  arg[iarg].u.s = static_cast<char*>(emalloc(strlen(buf) + 1));
350  arglist[iarg] = (void*) arg[iarg].u.s;
351  break;
352  case 'c':

354  goto bad_specifier;
355  }

358  break;
359  default:
360  goto bad_specifier;
361  /*break;*/
362  }
363  if (convert) {
364  ++iarg;
365  if (!ifarg(iarg + 2)) {
366  goto missing_arg;
367  }
368  switch (arg[iarg - 1].type) {
369  case 's':
370  if (!hoc_is_str_arg(iarg + 2)) {
371  Sprintf(errbuf, "arg %d must be a string", iarg + 2);
372  goto bad_arg;
373  }
374  break;
375  default:
376  if (!hoc_is_pdouble_arg(iarg + 2)) {
377  Sprintf(errbuf, "arg %d must be a pointer to a number", iarg + 2);
378  goto bad_arg;
379  }
380  break;
381  }
382  }
383  }
384  }
385 
386  if (iarg < 4) {
387  n = sscanf(buf, format, arglist[0], arglist[1], arglist[2]);
388  } else if (iarg < 13) {
389  n = sscanf(buf,
390  format,
391  arglist[0],
392  arglist[1],
393  arglist[2],
394  arglist[3],
395  arglist[4],
396  arglist[5],
397  arglist[6],
398  arglist[7],
399  arglist[8],
400  arglist[9],
401  arglist[10],
402  arglist[11]);
403  } else {
404  goto too_many;
405  }
406  assert(n <= iarg);
407 
408  for (i = 0; i < n; ++i) {
409  switch (arg[i].type) {
410  case 'd':
411  *hoc_pgetarg(i + 3) = arg[i].u.d;
412  break;
413  case 'i':
414  *hoc_pgetarg(i + 3) = (double) arg[i].u.i;
415  break;
416  case 'l':
417  *hoc_pgetarg(i + 3) = (double) arg[i].u.l;
418  break;
419  case 'f':
420  *hoc_pgetarg(i + 3) = (double) arg[i].u.f;
421  break;
422  case 's':
423  hoc_assign_str(hoc_pgargstr(i + 3), arg[i].u.s);
424  break;
425  case 'c':
426  *hoc_pgetarg(i + 3) = (double) arg[i].u.c;
427  break;
428  }
429  }
430  goto normal;
431 incomplete:
432  Sprintf(errbuf, "incomplete format specifier for arg %d", iarg + 3);
433  goto normal;
434 bad_specifier:
435  Sprintf(errbuf, "unknown conversion specifier for arg %d", iarg + 3);
436  goto normal;
437 missing_arg:

439  goto normal;
440 bad_arg:
441  goto normal;
442 too_many:

444  goto normal;
445 normal:

447  if (arg[i].type == 's') {
448  free(arg[i].u.s);
449  }
450  }
451  if (errbuf[0]) {
452  hoc_execerror("scan error:", errbuf);
453  }
454  return n;
455 }
