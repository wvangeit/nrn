73 /**
74  * Convert char* containing arguments from neuron to char* argv[] for
75  * coreneuron command line argument parser.
76  */
77 char* prepare_args(int& argc, char**& argv, std::string& args) {
78  // we can't modify string with strtok, make copy
79  char* first = strdup(args.c_str());
80  const char* sep = " ";
81 
82  // first count the no of argument
83  char* token = strtok(first, sep);
84  argc = 0;
85  while (token) {
86  token = strtok(nullptr, sep);
87  argc++;
88  }
89  free(first);
90 
91  // now build char*argv
92  argv = new char*[argc + 1];
93  first = strdup(args.c_str());
94  token = strtok(first, sep);
95  for (int i = 0; token; i++) {
96  argv[i] = token;
97  token = strtok(nullptr, sep);
98  }
99 
100  // make sure argv is terminated by NULL!
101  argv[argc] = nullptr;
102 
103  // return actual data to be freed
104  return first;
105 }

111 // bsize = 0 then per step transfer
112 // bsize > 1 then full trajectory save into arrays.
113 void get_nrn_trajectory_requests(int bsize) {
114  if (nrn2core_get_trajectory_requests_) {
115  for (int tid = 0; tid < nrn_nthread; ++tid) {
116  NrnThread& nt = nrn_threads[tid];
117  int n_pr;
118  int n_trajec;
119  int* types;
120  int* indices;
121  void** vpr;
122  double** varrays;
123  double** pvars;
124 
125  // bsize is passed by reference, the return value will determine if
126  // per step return or entire trajectory return.
127  (*nrn2core_get_trajectory_requests_)(
128  tid, bsize, n_pr, vpr, n_trajec, types, indices, pvars, varrays);
129  delete_trajectory_requests(nt);
130  if (n_trajec) {
131  TrajectoryRequests* tr = new TrajectoryRequests;
132  nt.trajec_requests = tr;
133  tr->bsize = bsize;
134  tr->n_pr = n_pr;
135  tr->n_trajec = n_trajec;
136  tr->vsize = 0;
137  tr->vpr = vpr;
138  tr->gather = new double*[n_trajec];
139  tr->varrays = varrays;
140  tr->scatter = pvars;
141  for (int i = 0; i < n_trajec; ++i) {
142  tr->gather[i] = stdindex2ptr(types[i], indices[i], nt);
143  }
144  delete[] types;
145  delete[] indices;
146  }
147  }
148  }

203 
204  // set global variables for start time, timestep and temperature
205  if (!corenrn_embedded) {
206  t = checkPoints.restore_time();
207  }
208 
209  if (corenrn_param.dt != -1000.) { // command line arg highest precedence
210  dt = corenrn_param.dt;
211  } else if (dt == -1000.) { // not on command line and no dt in globals.dat
212  dt = 0.025; // lowest precedence
213  }
214 
215  corenrn_param.dt = dt;
216 
217  rev_dt = (int) (1. / dt);
218 
219  if (corenrn_param.celsius != -1000.) { // command line arg highest precedence
220  celsius = corenrn_param.celsius;
221  } else if (celsius == -1000.) { // not on command line and no celsius in globals.dat
222  celsius = 34.0; // lowest precedence
223  }
224 
225  corenrn_param.celsius = celsius;
226 
227  // create net_cvode instance
228  mk_netcvode();
229 
230  // One part done before call to nrn_setup. Other part after.
231 
232  if (!corenrn_param.patternstim.empty()) {

235

239

272 
273  if (!corenrn_param.is_quiet()) {
274  report_mem_usage("After nrn_setup ");

277  // Invoke PatternStim
278  if (!corenrn_param.patternstim.empty()) {
279  nrn_mkPatternStim(corenrn_param.patternstim.c_str(), corenrn_param.tstop);
280  }
281 
282  /// Setting the timeout
283  nrn_set_timeout(200.);
284 
285  // show all configuration parameters for current run
286  if (nrnmpi_myid == 0 && !corenrn_param.is_quiet()) {
287  std::cout << corenrn_param << std::endl;
288  std::cout << " Start time (t) = " << t << std::endl << std::endl;
289  }
290 
291  // allocate buffer for mpi communication
292  mk_spikevec_buffer(corenrn_param.spikebuf);
293 
294  if (!corenrn_param.is_quiet()) {
295  report_mem_usage("After mk_spikevec_buffer");
296  }
297 
298  // In direct mode there are likely trajectory record requests
299  // to allow processing in NEURON after simulation by CoreNEURON
300  if (corenrn_embedded) {
301  // arg is additional vector size required (how many items will be
302  // written to the double*) but NEURON can instead
303  // specify that returns will be on a per time step basis.
304  get_nrn_trajectory_requests(int((corenrn_param.tstop - t) / corenrn_param.dt) + 2);
305 
306  // In direct mode, CoreNEURON has exactly the behavior of
307  // ParallelContext.psolve(tstop). Ie a sequence of such calls
308  // without an intervening h.finitialize() continues from the end
309  // of the previous call. I.e., all initial state, including
310  // the event queue has been set up in NEURON. And, at the end
311  // all final state, including the event queue will be sent back
312  // to NEURON. Here there is some first time only

328  allocate_data_in_mechanism_nrn_init();

332  if (nrn_have_gaps) {
333  nrn_partrans::copy_gap_indices_to_device();

335  }

338  call_prcellstate_for_prcellgid(corenrn_param.prcellgid, corenrn_param.gpu, 1);

355 /* perform forwardskip and call prcellstate for prcellgid */
356 void handle_forward_skip(double forwardskip, int prcellgid) {
357  double savedt = dt;
358  double savet = t;
359 
360  dt = forwardskip * 0.1;
361  t = -1e9;
362  dt2thread(-1.);
363 
364  for (int step = 0; step < 10; ++step) {
365  nrn_fixed_step_minimal();
366  }
367 
368  if (prcellgid >= 0) {
369  prcellstate(prcellgid, "fs");
370  }
371 
372  dt = savedt;
373  t = savet;
374  dt2thread(-1.);
375 
376  // clear spikes generated during forward skip (with negative time)
377  clear_spike_vectors();
378 }

490 extern "C" int run_solve_core(int argc, char** argv) {
491  Instrumentor::phase_begin("main");
492 
493  std::vector<ReportConfiguration> configs;
494  std::vector<std::unique_ptr<ReportHandler>> report_handlers;
495  SpikesInfo spikes_info;
496  bool reports_needs_finalize = false;
497 
498  if (!corenrn_param.is_quiet()) {
499  report_mem_usage("After mk_mech");
500  }
501 
502  // Create outpath if it does not exist
503  if (nrnmpi_myid == 0) {
504  fs::create_directories(corenrn_param.outpath);
505  }
506 
507  if (!corenrn_param.reportfilepath.empty()) {
508  configs = create_report_configurations(corenrn_param.reportfilepath,
509  corenrn_param.outpath,
510  spikes_info);
511  reports_needs_finalize = !configs.empty();
512  }
513 
514  CheckPoints checkPoints{corenrn_param.checkpointpath, corenrn_param.restorepath};
515 
516  // initializationa and loading functions moved to separate
517  {
518  Instrumentor::phase p("load-model");
519  nrn_init_and_load_data(argc, argv, checkPoints, !configs.empty());
520  }
521 
522  std::string output_dir = corenrn_param.outpath;
523 
524  if (nrnmpi_myid == 0) {
525  fs::create_directories(output_dir);
526  }
527 #if NRNMPI
528  if (corenrn_param.mpi_enable) {
529  nrnmpi_barrier();
530  }
531 #endif
532  bool compute_gpu = corenrn_param.gpu;
533 
534  nrn_pragma_acc(update device(celsius, secondorder, pi) if (compute_gpu))
535  nrn_pragma_omp(target update to(celsius, secondorder, pi) if (compute_gpu))
536  {
537  double v = corenrn_param.voltage;
538  double dt = corenrn_param.dt;
539  double delay = corenrn_param.mindelay;
540  double tstop = corenrn_param.tstop;
541 
542  if (tstop < t && nrnmpi_myid == 0) {
543  printf("Error: Stop time (%lf) < Start time (%lf), restoring from checkpoint? \n",
544  tstop,
545  t);
546  abort();
547  }
548 
549  // TODO : if some ranks are empty then restore will go in deadlock
550  // phase (as some ranks won't have restored anything and hence return
551  // false in checkpoint_initialize
552  if (!corenrn_embedded && !checkPoints.initialize()) {
553  nrn_finitialize(v != 1000., v);
554  }
555 
556  if (!corenrn_param.is_quiet()) {
557  report_mem_usage("After nrn_finitialize");
558  }
559 
560  // register all reports with libsonata
561  double min_report_dt = INT_MAX;
562  for (size_t i = 0; i < configs.size(); i++) {
563  std::unique_ptr<ReportHandler> report_handler = create_report_handler(configs[i],
564  spikes_info);
565  if (report_handler) {
566  report_handler->create_report(configs[i], dt, tstop, delay);
567  report_handlers.push_back(std::move(report_handler));
568  }
569  if (configs[i].report_dt < min_report_dt) {
570  min_report_dt = configs[i].report_dt;
571  }
572  }
573  // Set the buffer size if is not the default value. Otherwise use report.conf on
574  // register_report
575  if (corenrn_param.report_buff_size != corenrn_param.report_buff_size_default) {
576  set_report_buffer_size(corenrn_param.report_buff_size);
577  }
578 
579  if (!configs.empty()) {
580  setup_report_engine(min_report_dt, delay);
581  configs.clear();
582  }
583 
584  // call prcellstate for prcellgid
585  call_prcellstate_for_prcellgid(corenrn_param.prcellgid, compute_gpu, 0);
586 
587  // handle forwardskip
588  if (corenrn_param.forwardskip > 0.0) {
589  Instrumentor::phase p("handle-forward-skip");
590  handle_forward_skip(corenrn_param.forwardskip, corenrn_param.prcellgid);
591  }
592 
593  /// Solver execution
594  Instrumentor::start_profile();
595  Instrumentor::phase_begin("simulation");
596  BBS_netpar_solve(corenrn_param.tstop);
597  Instrumentor::phase_end("simulation");
598  Instrumentor::stop_profile();
599 
600  // update cpu copy of NrnThread from GPU
601  update_nrnthreads_on_host(nrn_threads, nrn_nthread);
602 
603  // direct mode and full trajectory gathering on CoreNEURON, send back.
604  if (corenrn_embedded) {
605  trajectory_return();
606  }
607 
608  // Report global cell statistics
609  if (!corenrn_param.is_quiet()) {
610  report_cell_stats();
611  }
612 
613  // prcellstate after end of solver
614  call_prcellstate_for_prcellgid(corenrn_param.prcellgid, compute_gpu, 0);
615  }
616 
617  // write spike information to outpath
618  {
619  Instrumentor::phase p("output-spike");
620  output_spikes(output_dir.c_str(), spikes_info);
621  }
622 
623  // copy weights back to NEURON NetCon
624  if (nrn2core_all_weights_return_ && corenrn_embedded) {
625  // first update weights from gpu
626  update_weights_from_gpu(nrn_threads, nrn_nthread);
627 
628  // store weight pointers
629  std::vector<double*> weights(nrn_nthread, nullptr);
630 
631  // could be one thread more (empty) than in NEURON but does not matter
632  for (int i = 0; i < nrn_nthread; ++i) {
633  weights[i] = nrn_threads[i].weights;
634  }
635  (*nrn2core_all_weights_return_)(weights);
636  }
637 
638  if (corenrn_embedded) {
639  core2nrn_data_return();
640  }
