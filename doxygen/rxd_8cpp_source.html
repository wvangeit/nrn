500  treesolver
501 
502  a - above the diagonal
503  b - below the diagonal
504  c - used to define diagonal: diag = c + dt * dbase
505  dbase - together with c, used to define diagonal
506  rhs - right hand side, which is changed to the result
507  pindex - parent indices
508  n - number of states
509  */

519  myd = d;

521  myc = c;
522  for (i = 0; i < n; i++) {
523  *myd++ = *myc++ + dt * (*mydbase++);
524  }
525

527  for (i = n - 1; i > 0; --i) {
528  pin = pindex[i];
529  if (pin > -1) {
530  double p;

532  d[pin] -= dt * p * b[i];

538  pin = pindex[i];
539  if (pin > -1) {
540  rhs[i] -= dt * b[i] * rhs[pin];
541  }
542  rhs[i] /= d[i];
543  }
544

1095 void TaskQueue_exe_tasks(std::size_t thread_index, TaskQueue* q) {
1096  for (;;) {
1097  // Wait for a task to be available in the queue, then execute it.
1098  {
1099  TaskList* job{};
1100  {
1101  std::unique_lock<std::mutex> lock{q->task_mutex};
1102  // Wait until either for a new task to be received or for this
1103  // thread to be told to exit.
1104  q->task_cond.wait(lock,
1105  [q, thread_index] { return q->first || q->exit[thread_index]; });
1106  if (q->exit[thread_index]) {
1107  return;
1108  }
1109  job = q->first;
1110  q->first = job->next;
1111  }
1112  // Execute the task
1113  job->result = job->task(job->args);
1114  delete job;
1115  }
1116  // Decrement the list length, if it's now empty then broadcast that to
1117  // the master thread, which may be waiting for the queue to be empty.
1118  auto const new_length = [q] {
1119  std::lock_guard<std::mutex> _{q->waiting_mutex};
1120  return --(q->length);
1121  }();
1122  // The immediately-executed lambda means we release the mutex before
1123  // calling notify_one()
1124  if (new_length == 0) {
1125  // Queue is empty. Notify the main thread, which may be blocking on
1126  // this condition.
1127  q->waiting_cond.notify_one();
1128  }
1129  }
1130 }

1133 void set_num_threads(const int n) {
1134  assert(n > 0);
1135  assert(NUM_THREADS > 0);
1136  // n and NUM_THREADS include the main thread, old_num and new_num refer to
1137  // the number of std::thread workers
1138  std::size_t const old_num = NUM_THREADS - 1;
1139  std::size_t const new_num = n - 1;
1140  assert(old_num == Threads.size());
1141  assert(old_num == task_queue.exit.size());
1142  if (new_num < old_num) {
1143  // Kill some threads. First, wait until the queue is empty.
1144  TaskQueue_sync(&task_queue);
1145  // Now signal to the threads that are to be killed that they need to
1146  // exit.
1147  {
1148  std::lock_guard<std::mutex> _{task_queue.task_mutex};
1149  for (auto k = new_num; k < old_num; ++k) {
1150  task_queue.exit[k] = true;
1151  }
1152  }
1153  task_queue.task_cond.notify_all();
1154  // Finally, join those threads and destroy the std::thread objects.
1155  for (auto k = new_num; k < old_num; ++k) {
1156  Threads[k].join();
1157  }
1158  // And update the structures
1159  {
1160  std::lock_guard<std::mutex> _{task_queue.task_mutex};
1161  Threads.resize(new_num);
1162  task_queue.exit.resize(new_num);
1163  }
1164  } else if (new_num > old_num) {
1165  // Create some threads
1166  std::lock_guard<std::mutex> _{task_queue.task_mutex};
1167  task_queue.exit.reserve(new_num);
1168  Threads.reserve(new_num);
1169  for (auto k = old_num; k < new_num; ++k) {
1170  assert(k == Threads.size());
1171  Threads.emplace_back(TaskQueue_exe_tasks, k, &task_queue);
1172  task_queue.exit.emplace_back(false);
1173  }
1174  }
1175  assert(new_num == Threads.size());
1176  assert(new_num == task_queue.exit.size());
1177  set_num_threads_3D(n);
1178  NUM_THREADS = n;
1179 }
