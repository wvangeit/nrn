27 enum struct Transform { None, ViaRawPointer, ViaGenericDataHandle };
28 template <typename T>
29 data_handle<T> transform(data_handle<T> handle, Transform type) {
30  if (type == Transform::None) {
31  return handle;
32  } else if (type == Transform::ViaRawPointer) {
33  return data_handle<T>{static_cast<T*>(handle)};
34  } else {
35  assert(type == Transform::ViaGenericDataHandle);
36  return static_cast<data_handle<T>>(generic_data_handle{handle});
37  }
38 }
