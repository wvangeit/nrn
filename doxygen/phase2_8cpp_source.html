89 template <typename T>
90 inline void mech_data_layout_transform(T* data, int cnt, int sz, int layout) {
91  if (layout == Layout::AoS) {
92  return;
93  }
94  // layout is equal to Layout::SoA
95  int align_cnt = nrn_soa_padded_size(cnt, layout);
96  std::vector<T> d(cnt * sz);
97  // copy matrix
98  for (int i = 0; i < cnt; ++i) {
99  for (int j = 0; j < sz; ++j) {
100  d[i * sz + j] = data[i * sz + j];
101  }
102  }
103  // transform memory layout
104  for (int i = 0; i < cnt; ++i) {
105  for (int j = 0; j < sz; ++j) {
106  data[i + j * align_cnt] = d[i * sz + j];
107  }
108  }
109 }

532  int type;

534  double time;
535  F.read_array(&time, 1);
536  switch (type) {
537  case NetConType: {
538  auto event = std::make_shared<NetConType_>();
539  event->time = time;
540  event->netcon_index = F.read_int();
541  events.emplace_back(type, event);
542  break;
543  }
544  case SelfEventType: {
545  auto event = std::make_shared<SelfEventType_>();
546  event->time = time;
547  event->target_type = F.read_int();
548  event->point_proc_instance = F.read_int();
549  event->target_instance = F.read_int();
550  F.read_array(&event->flag, 1);
551  event->movable = F.read_int();
552  event->weight_index = F.read_int();
553  events.emplace_back(type, event);
554  break;
555  }
556  case PreSynType: {
557  auto event = std::make_shared<PreSynType_>();
558  event->time = time;
559  event->presyn_index = F.read_int();
560  events.emplace_back(type, event);
561  break;
562  }
563  case NetParEventType: {
564  auto event = std::make_shared<NetParEvent_>();
565  event->time = time;
566  events.emplace_back(type, event);

574  event->vecplay_index = F.read_int();
575  events.emplace_back(type, event);

577  nrn_assert(0);

580  }

584  }
585  }

622 void Phase2::pdata_relocation(const NrnThread& nt, const std::vector<Memb_func>& memb_func) {
623  // Some pdata may index into data which has been reordered from AoS to
624  // SoA. The four possibilities are if semantics is -1 (area), -5 (pointer),
625  // -9 (diam), // or 0-999 (ion variables).
626  // Note that pdata has a layout and the // type block in nt.data into which
627  // it indexes, has a layout.
628 
629  // For faster search of tmls[i].type == type, use a map.
630  // (perhaps would be better to replace tmls so that we can use tmls[type].
631  std::map<int, size_t> type2itml;
632  for (size_t i = 0; i < tmls.size(); ++i) {
633  if (tmls[i].pointer2type.size()) {
634  type2itml[tmls[i].type] = i;
635  }
636  }
637 
638  for (auto tml = nt.tml; tml; tml = tml->next) {
639  int type = tml->index;
640  int layout = corenrn.get_mech_data_layout()[type];
641  int* pdata = tml->ml->pdata;
642  int cnt = tml->ml->nodecount;
643  int szdp = corenrn.get_prop_dparam_size()[type];
644  int* semantics = memb_func[type].dparam_semantics;
645 
646  // compute only for ARTIFICIAL_CELL (has useful area pointer with semantics=-1)
647  if (!corenrn.get_is_artificial()[type]) {
648  if (szdp) {
649  if (!semantics)
650  continue; // temporary for HDFReport, Binreport which will be skipped in
651  // bbcore_write of HBPNeuron
652  nrn_assert(semantics);
653  }
654 
655  for (int i = 0; i < szdp; ++i) {
656  int s = semantics[i];
657  switch (s) {
658  case -1: // area
659  transform_int_data(
660  nt._actual_area - nt._data, cnt, pdata, i, szdp, layout, nt.end);
661  break;
662  case -9: // diam
663  transform_int_data(
664  nt._actual_diam - nt._data, cnt, pdata, i, szdp, layout, nt.end);
665  break;
666  case -5: // pointer assumes a pointer to membrane voltage
667  // or mechanism data in this thread. The value of the
668  // pointer on the NEURON side was analyzed by
669  // nrn_dblpntr2nrncore which returned the
670  // mechanism index and type. At this moment the index
671  // is in pdata and the type is in tmls[type].pointer2type.
672  // However the latter order is according to the nested
673  // iteration for nodecount { for szdp {}}
674  // Also the nodecount POINTER instances of mechanism
675  // might possibly point to differnt range variables.
676  // Therefore it is not possible to use transform_int_data
677  // and the transform must be done one at a time.
678  // So we do nothing here and separately iterate
679  // after this loop instead of the former voltage only
680  /**
681  transform_int_data(
682  nt._actual_v - nt._data, cnt, pdata, i, szdp, layout, nt.end);
683  **/
684  break;
685  default:
686  if (s >= 0 && s < 1000) { // ion
687  int etype = s;
688  /* if ion is SoA, must recalculate pdata values */
689  /* if ion is AoS, have to deal with offset */
690  Memb_list* eml = nt._ml_list[etype];
691  int edata0 = eml->data - nt._data;
692  int ecnt = eml->nodecount;
693  int esz = corenrn.get_prop_param_size()[etype];
694  for (int iml = 0; iml < cnt; ++iml) {
695  int* pd = pdata + nrn_i_layout(iml, cnt, i, szdp, layout);
696  int ix = *pd; // relative to the ion data
697  nrn_assert((ix >= 0) && (ix < ecnt * esz));
698  /* Original pd order assumed ecnt groups of esz */
699  *pd = edata0 + nrn_param_layout(ix, etype, eml);
700  }
701  }
702  }
703  }
704  // Handle case -5 POINTER transformation (see comment above)
705  auto search = type2itml.find(type);
706  if (search != type2itml.end()) {
707  auto& ptypes = tmls[type2itml[type]].pointer2type;
708  assert(ptypes.size());
709  size_t iptype = 0;
710  for (int iml = 0; iml < cnt; ++iml) {
711  for (int i = 0; i < szdp; ++i) {
712  if (semantics[i] == -5) { // POINTER
713  int* pd = pdata + nrn_i_layout(iml, cnt, i, szdp, layout);
714  int ix = *pd; // relative to elem0
715  int ptype = ptypes[iptype++];
716  if (ptype == voltage) {
717  nrn_assert((ix >= 0) && (ix < nt.end));
718  int elem0 = nt._actual_v - nt._data;
719  *pd = elem0 + ix;
720  } else {
721  Memb_list* pml = nt._ml_list[ptype];
722  int pcnt = pml->nodecount;
723  int psz = corenrn.get_prop_param_size()[ptype];
724  nrn_assert((ix >= 0) && (ix < pcnt * psz));
725  int elem0 = pml->data - nt._data;
726  *pd = elem0 + nrn_param_layout(ix, ptype, pml);
727  }
728  }
729  }
730  }
731  ptypes.clear();
732  }
733  }

737 void Phase2::set_dependencies(const NrnThread& nt, const std::vector<Memb_func>& memb_func) {
738  /* here we setup the mechanism dependencies. if there is a mechanism dependency
739  * then we allocate an array for tml->dependencies otherwise set it to nullptr.
740  * In order to find out the "real" dependencies i.e. dependent mechanism
741  * exist at the same compartment, we compare the nodeindices of mechanisms
742  * returned by nrn_mech_depend.
743  */
744 
745  /* temporary array for dependencies */
746  int* mech_deps = (int*) ecalloc(memb_func.size(), sizeof(int));
747 
748  for (auto tml = nt.tml; tml; tml = tml->next) {
749  /* initialize to null */
750  tml->dependencies = nullptr;
751  tml->ndependencies = 0;
752 
753  /* get dependencies from the models */
754  int deps_cnt = nrn_mech_depend(tml->index, mech_deps);
755 
756  /* if dependencies, setup dependency array */
757  if (deps_cnt) {
758  /* store "real" dependencies in the vector */
759  std::vector<int> actual_mech_deps;
760 
761  Memb_list* ml = tml->ml;
762  int* nodeindices = ml->nodeindices;
763 
764  /* iterate over dependencies */
765  for (int j = 0; j < deps_cnt; j++) {
766  /* memb_list of dependency mechanism */
767  Memb_list* dml = nt._ml_list[mech_deps[j]];
768 
769  /* dependency mechanism may not exist in the model */
770  if (!dml)
771  continue;
772 
773  /* take nodeindices for comparison */
774  int* dnodeindices = dml->nodeindices;
775 
776  /* set_intersection function needs temp vector to push the common values */
777  std::vector<int> node_intersection;
778 
779  /* make sure they have non-zero nodes and find their intersection */
780  if ((ml->nodecount > 0) && (dml->nodecount > 0)) {
781  std::set_intersection(nodeindices,
782  nodeindices + ml->nodecount,
783  dnodeindices,
784  dnodeindices + dml->nodecount,
785  std::back_inserter(node_intersection));
786  }
787 
788  /* if they intersect in the nodeindices, it's real dependency */
789  if (!node_intersection.empty()) {
790  actual_mech_deps.push_back(mech_deps[j]);
791  }
792  }
793 
794  /* copy actual_mech_deps to dependencies */
795  if (!actual_mech_deps.empty()) {
796  tml->ndependencies = actual_mech_deps.size();
797  tml->dependencies = (int*) ecalloc(actual_mech_deps.size(), sizeof(int));
798  std::copy(actual_mech_deps.begin(), actual_mech_deps.end(), tml->dependencies);
799  }
800  }
801  }
802 
803  /* free temp dependency array */
804  free(mech_deps);
805 }
