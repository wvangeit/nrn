210  netsend, pointer, pntproc, bbcorepointer, watch, diam, fornetcon

231  } else if (strcmp(name, "watch") == 0) {
232  memb_func[type].dparam_semantics[ix] = -8;

275 static int depend_append(int idep, int* dependencies, int deptype, int type) {
276  /* append only if not already in dependencies and != type*/
277  bool add = true;
278  if (deptype == type) {
279  return idep;
280  }
281  for (int i = 0; i < idep; ++i) {
282  if (deptype == dependencies[i]) {
283  add = false;
284  break;
285  }
286  }
287  if (add) {
288  dependencies[idep++] = deptype;
289  }
290  return idep;
291 }

293 /* return list of types that this type depends on (10 should be more than enough) */
294 /* dependencies must be an array that is large enough to hold that array */
295 /* number of dependencies is returned */
296 int nrn_mech_depend(int type, int* dependencies) {
297  int dpsize = corenrn.get_prop_dparam_size()[type];
298  int* ds = corenrn.get_memb_func(type).dparam_semantics;
299  int idep = 0;
300  if (ds)
301  for (int i = 0; i < dpsize; ++i) {
302  if (ds[i] > 0 && ds[i] < 1000) {
303  int deptype = ds[i];
304  int idepnew = depend_append(idep, dependencies, deptype, type);
305  if ((idepnew > idep) && !corenrn.get_ion_write_dependency().empty() &&
306  !corenrn.get_ion_write_dependency()[deptype].empty()) {
307  auto& iwd = corenrn.get_ion_write_dependency()[deptype];
308  int size = iwd[0];
309  for (int j = 1; j < size; ++j) {
310  idepnew = depend_append(idepnew, dependencies, iwd[j], type);
311  }
312  }
313  idep = idepnew;
314  }
315  }
316  return idep;
317 }
