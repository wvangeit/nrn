307 /* Complex division: to = num / den */
308 #define CMPLX_DIV(to, num, den) \
309  { \
310  RealNumber r_, s_; \
311  if (((den).Real >= (den).Imag AND(den).Real > -(den).Imag) OR((den).Real < (den).Imag AND(den).Real <= -(den).Imag)) { \
312  r_ = (den).Imag / (den).Real; \
313  s_ = (den).Real + r_ * (den).Imag; \
314  (to).Real = ((num).Real + r_ * (num).Imag) / s_; \
315  (to).Imag = ((num).Imag - r_ * (num).Real) / s_; \
316  } else { \
317  r_ = (den).Real / (den).Imag; \
318  s_ = (den).Imag + r_ * (den).Real; \
319  (to).Real = (r_ * (num).Real + (num).Imag) / s_; \
320  (to).Imag = (r_ * (num).Imag - (num).Real) / s_; \
321  } \
322  }
323 
324 /* Complex division and assignment: num /= den */
325 #define CMPLX_DIV_ASSIGN(num, den) \
326  { \
327  RealNumber r_, s_, t_; \
328  if (((den).Real >= (den).Imag AND(den).Real > -(den).Imag) OR((den).Real < (den).Imag AND(den).Real <= -(den).Imag)) { \
329  r_ = (den).Imag / (den).Real; \
330  s_ = (den).Real + r_ * (den).Imag; \
331  t_ = ((num).Real + r_ * (num).Imag) / s_; \
332  (num).Imag = ((num).Imag - r_ * (num).Real) / s_; \
333  (num).Real = t_; \
334  } else { \
335  r_ = (den).Real / (den).Imag; \
336  s_ = (den).Imag + r_ * (den).Real; \
337  t_ = (r_ * (num).Real + (num).Imag) / s_; \
338  (num).Imag = (r_ * (num).Imag - (num).Real) / s_; \
339  (num).Real = t_; \
340  } \
341  }

458 /*
459  * MATRIX ELEMENT DATA STRUCTURE
460  *
461  * Every nonzero element in the matrix is stored in a dynamically allocated
462  * MatrixElement structure. These structures are linked together in an
463  * orthogonal linked list. Two different MatrixElement structures exist.
464  * One is used when only real matrices are expected, it is missing an entry
465  * for imaginary data. The other is used if complex matrices are expected.
466  * It contains an entry for imaginary data.
467  *
468  * >>> Structure fields:
469  * Real (RealNumber)
470  * The real portion of the value of the element. Real must be the first
471  * field in this structure.
472  * Imag (RealNumber)
473  * The imaginary portion of the value of the element. If the matrix
474  * routines are not compiled to handle complex matrices, then this
475  * field does not exist. If it exists, it must follow immediately after
476  * Real.
477  * Row (int)
478  * The row number of the element.
479  * Col (int)
480  * The column number of the element.
481  * NextInRow (struct MatrixElement *)
482  * NextInRow contains a pointer to the next element in the row to the
483  * right of this element. If this element is the last nonzero in the
484  * row then NextInRow contains NULL.
485  * NextInCol (struct MatrixElement *)
486  * NextInCol contains a pointer to the next element in the column below
487  * this element. If this element is the last nonzero in the column then
488  * NextInCol contains NULL.
489  * pInitInfo (char *)
490  * Pointer to user data used for initialization of the matrix element.
491  * Initialized to NULL.
492  *
493  * >>> Type definitions:
494  * ElementPtr
495  * A pointer to a MatrixElement.
496  * ArrayOfElementPtrs
497  * An array of ElementPtrs. Used for FirstInRow, FirstInCol and
498  * Diag pointer arrays.
499  */
500 
501 /* Begin `MatrixElement'. */
502

504  RealNumber Real;
505 #if spCOMPLEX
506  RealNumber Imag;
507 #endif
508  int Row;
509  int Col;
510  struct MatrixElement* NextInRow;
511  struct MatrixElement* NextInCol;
512 #if INITIALIZE
513  char* pInitInfo;
514 #endif
515 };

570 /*
571  * MATRIX FRAME DATA STRUCTURE
572  *
573  * This structure contains all the pointers that support the orthogonal
574  * linked list that contains the matrix elements. Also included in this
575  * structure are other numbers and pointers that are used globally by the
576  * sparse matrix routines and are associated with one particular matrix.
577  *
578  * >>> Type definitions:
579  * MatrixPtr
580  * A pointer to MatrixFrame. Essentially, a pointer to the matrix.
581  *
582  * >>> Structure fields:
583  * AbsThreshold (RealNumber)
584  * The absolute magnitude an element must have to be considered as a
585  * pivot candidate, except as a last resort.
586  * AllocatedExtSize (int)
587  * The allocated size of the arrays used to translate external row and
588  * column numbers to their internal values.
589  * AllocatedSize (int)
590  * The currently allocated size of the matrix; the size the matrix can
591  * grow to when EXPANDABLE is set true and AllocatedSize is the largest
592  * the matrix can get without requiring that the matrix frame be
593  * reallocated.
594  * Complex (BOOLEAN)
595  * The flag which indicates whether the matrix is complex (true) or
596  * real.
597  * CurrentSize (int)
598  * This number is used during the building of the matrix when the
599  * TRANSLATE option is set true. It indicates the number of internal
600  * rows and columns that have elements in them.
601  * Diag (ArrayOfElementPtrs)
602  * Array of pointers that points to the diagonal elements.
603  * DoCmplxDirect (BOOLEAN *)
604  * Array of flags, one for each column in matrix. If a flag is true
605  * then corresponding column in a complex matrix should be eliminated
606  * in spFactor() using direct addressing (rather than indirect
607  * addressing).
608  * DoRealDirect (BOOLEAN *)
609  * Array of flags, one for each column in matrix. If a flag is true
610  * then corresponding column in a real matrix should be eliminated
611  * in spFactor() using direct addressing (rather than indirect
612  * addressing).
613  * Elements (int)
614  * The number of original elements (total elements minus fill ins)
615  * present in matrix.
616  * Error (int)
617  * The error status of the sparse matrix package.
618  * ExtSize (int)
619  * The value of the largest external row or column number encountered.
620  * ExtToIntColMap (int [])
