7 Now that threads have taken over the actual_v, v_node, etc, it might
8 be a good time to regularize the method of freeing, allocating, and
9 updating those arrays. To recapitulate the history, Node used to
10 be the structure that held direct values for v, area, d, rhs, etc.
11 That continued to hold for the cray vectorization project which
12 introduced v_node, v_parent, memb_list. Cache efficiency introduced
13 actual_v, actual_area, actual_d, etc and the Node started pointing
14 into those arrays. Additional nodes after allocation required updating
15 pointers to v and area since those arrays were freed and reallocated.
16 Now, the threads hold all these arrays and we want to update them
17 properly under the circumstances of changing topology, changing
18 number of threads, and changing distribution of cells on threads.
19 Note there are no longer global versions of any of these arrays.
20 We do not want to update merely due to a change in area. Recently
21 we have dealt with diam, area, ri on a section basis. We generally
22 desire an update just before a simulation when the efficient
23 structures are necessary. This is reasonably well handled by the
24 v_structure_change flag which historically freed and reallocated
25 v_node and v_parent and, just before this comment,
26 ended up setting the NrnThread tml. This makes most of the old
27 memb_list vestigial and we now got rid of it except for
28 the artificial cells (and it is possibly not really necessary there).
29 Switching between sparse and tree matrix just cause freeing and
30 reallocation of actual_rhs.

32 If we can get the freeing, reallocation, and pointer update correct
33 for _actual_v, I am guessing everything else can be dragged along with
34 it. We have two major cases, call to pc.nthread and change in

127 void worker_main(worker_conf_t* my_wc_ptr,
128  std::condition_variable* my_cond_ptr,
129  std::mutex* my_mut_ptr) {
130  assert(my_cond_ptr);
131  assert(my_mut_ptr);
132  assert(my_wc_ptr);
133  auto& cond{*my_cond_ptr};
134  auto& mut{*my_mut_ptr};
135  auto& wc{*my_wc_ptr};
136  for (;;) {
137  if (busywait_) {
138  // WARNING: this branch has not been extensively tested after the
139  // std::thread migration.
140  while (wc.flag == worker_flag::wait) {
141  ;
142  }
143  if (wc.flag == worker_flag::exit) {
144  return;
145  }
146  assert(wc.flag == worker_flag::execute_job);
147  std::visit(worker_kernel{wc.thread_id}, wc.job);
148  wc.flag = worker_flag::wait;
149  wc.job = std::monostate{};
150  cond.notify_one();
151  } else {
152  worker_conf_t conf{};
153  {
154  // Wait until we have a job to execute or we have been told to
155  // shut down.
156  std::unique_lock<std::mutex> lock{mut};
157  cond.wait(lock, [&wc] { return wc.flag != worker_flag::wait; });
158  // Received instructions from the coordinator thread.
159  assert(wc.flag == worker_flag::execute_job || wc.flag == worker_flag::exit);
160  if (wc.flag == worker_flag::exit) {
161  return;
162  }
163  assert(wc.flag == worker_flag::execute_job);
164  // Save the workload + argument to local variables before
165  // releasing the mutex.
166  conf = wc;
167  }
168  // Execute the workload without keeping the mutex
169  std::visit(worker_kernel{conf.thread_id}, conf.job);
170  // Signal that the work is completed and this thread is becoming
171  // idle
172  {
173  std::lock_guard<std::mutex> _{mut};
174  // Make sure we don't accidentally overwrite an exit signal from
175  // the coordinating thread.
176  if (wc.flag == worker_flag::execute_job) {
177  assert(wc == conf);
178  wc.flag = worker_flag::wait;
179  wc.job = std::monostate{};
180  }
181  }
182  // Notify the coordinating thread.
183  cond.notify_one();
184  }
185  }
186 }

658 void reorder_secorder() {

661  Node* nd;
662  hoc_Item* qsec;
663  hoc_List* sl;
664  int order, isec, i, j, inode;
665  /* count and allocate */
666  // ForAllSections(sec)
667  ITERATE(qsec, section_list) {
668  Section* sec = hocSEC(qsec);
669  sec->order = -1;
670  }
671  order = 0;
672  FOR_THREADS(_nt) {
673  /* roots of this thread */
674  sl = _nt->roots;
675  inode = 0;
676  ITERATE(qsec, sl) {
677  sec = hocSEC(qsec);
678  assert(sec->order == -1);
679  secorder[order] = sec;
680  sec->order = order;
681  ++order;
682  nd = sec->parentnode;
683  nd->_nt = _nt;
684  inode += 1;
685  }
686  /* all children of what is already in secorder */
687  for (isec = order - _nt->ncell; isec < order; ++isec) {
688  sec = secorder[isec];
689  /* to make it easy to fill in PreSyn.nt_*/
690  sec->prop->dparam[9] = {neuron::container::do_not_search, _nt};
691  for (j = 0; j < sec->nnode; ++j) {
692  nd = sec->pnode[j];
693  nd->_nt = _nt;
694  inode += 1;
695  }
696  for (ch = sec->child; ch; ch = ch->sibling) {
697  assert(ch->order == -1);
698  secorder[order] = ch;
699  ch->order = order;
700  ++order;
701  }
702  }
703  _nt->end = inode;
704  CACHELINE_CALLOC(_nt->_v_node, Node*, inode);
705  CACHELINE_CALLOC(_nt->_v_parent, Node*, inode);
706  CACHELINE_CALLOC(_nt->_v_parent_index, int, inode);
707  }
708  /* do it again and fill _v_node and _v_parent */
709  /* index each cell section in relative order. Do offset later */
710  // ForAllSections(sec)
711  ITERATE(qsec, section_list) {
712  Section* sec = hocSEC(qsec);
713  sec->order = -1;
714  }
715  order = 0;
716  FOR_THREADS(_nt) {
717  /* roots of this thread */
718  sl = _nt->roots;
719  inode = 0;
720  ITERATE(qsec, sl) {
721  sec = hocSEC(qsec);
722  assert(sec->order == -1);
723  secorder[order] = sec;
724  sec->order = order;
725  ++order;
726  nd = sec->parentnode;
727  nd->_nt = _nt;
728  _nt->_v_node[inode] = nd;
729  _nt->_v_parent[inode] = nullptr; // because this is a root node
730  _nt->_v_node[inode]->v_node_index = inode;
731  ++inode;
732  }
733  /* all children of what is already in secorder */
734  for (isec = order - _nt->ncell; isec < order; ++isec) {
735  sec = secorder[isec];
736  /* to make it easy to fill in PreSyn.nt_*/
737  sec->prop->dparam[9] = {neuron::container::do_not_search, _nt};
738  for (j = 0; j < sec->nnode; ++j) {
739  nd = sec->pnode[j];
740  nd->_nt = _nt;
741  _nt->_v_node[inode] = nd;
742  if (j) {
743  _nt->_v_parent[inode] = sec->pnode[j - 1];
744  } else {
745  _nt->_v_parent[inode] = sec->parentnode;
746  }
747  _nt->_v_node[inode]->v_node_index = inode;
748  inode += 1;
749  }
750  for (ch = sec->child; ch; ch = ch->sibling) {
751  assert(ch->order == -1);
752  secorder[order] = ch;
753  ch->order = order;
754  ++order;
755  }
756  }
757  _nt->end = inode;
758  }
759  assert(order == section_count);
760  /*assert(inode == v_node_count);*/
761  /* not missing any */
762  // ForAllSections(sec)
763  ITERATE(qsec, section_list) {
764  Section* sec = hocSEC(qsec);
765  assert(sec->order != -1);
766  }
767

769  in either case, we can then point to v, d, rhs in proper
770  node order
771  */
772  FOR_THREADS(_nt) for (inode = 0; inode < _nt->end; ++inode) {
773  _nt->_v_node[inode]->_classical_parent = _nt->_v_parent[inode];
774  }
775  if (nrn_multisplit_setup_) {
776  /* classical order abandoned */
777  (*nrn_multisplit_setup_)();
778  }
779  /* because the d,rhs changed, if multisplit is used we need to update
780  the reduced tree gather/scatter pointers
781  */
782  if (nrn_multisplit_setup_) {
783  nrn_multisplit_ptr_update();
784  }
