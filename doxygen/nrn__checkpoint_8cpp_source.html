97 static int nrn_original_aos_index(int etype, int ix, NrnThread& nt, int** ml_pinv) {
98  // Determine ei_instance and ei from etype and ix.
99  // Deal with existing permutation and SoA.
100  Memb_list* eml = nt._ml_list[etype];
101  int ecnt = eml->nodecount;
102  int esz = corenrn.get_prop_param_size()[etype];
103  int elayout = corenrn.get_mech_data_layout()[etype];
104  // current index into eml->data is a function
105  // of elayout, eml._permute, ei_instance, ei, and
106  // eml padding.
107  int p = ix - (eml->data - nt._data);
108  assert(p >= 0 && p < eml->_nodecount_padded * esz);
109  int ei_instance, ei;
110  nrn_inverse_i_layout(p, ei_instance, ecnt, ei, esz, elayout);
111  if (elayout == Layout::SoA) {
112  if (eml->_permute) {
113  if (!ml_pinv[etype]) {
114  ml_pinv[etype] = inverse_permute(eml->_permute, eml->nodecount);
115  }
116  ei_instance = ml_pinv[etype][ei_instance];
117  }
118  }
119  return ei_instance * esz + ei;
120 }

652 T* CheckPoints::soa2aos(T* data, int cnt, int sz, int layout, int* permute) const {
653  // inverse of F -> data. Just a copy if layout=1. If SoA,
654  // original file order depends on padding and permutation.
655  // Good for a, b, area, v, diam, Memb_list.data, or anywhere values do not change.
656  T* d = new T[cnt * sz];
657  if (layout == Layout::AoS) {
658  for (int i = 0; i < cnt * sz; ++i) {
659  d[i] = data[i];
660  }
661  } else if (layout == Layout::SoA) {
662  int align_cnt = nrn_soa_padded_size(cnt, layout);
663  for (int i = 0; i < cnt; ++i) {
664  int ip = i;
665  if (permute) {
666  ip = permute[i];
667  }
668  for (int j = 0; j < sz; ++j) {
669  d[i * sz + j] = data[ip + j * align_cnt];
670  }
671  }
672  }
673  return d;
674 }
