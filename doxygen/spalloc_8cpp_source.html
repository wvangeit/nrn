78  * Allocates and initializes the data structures associated with a matrix.
79  *
80  * >>> Returned:
81  * A pointer to the matrix is returned cast into the form of a pointer to
82  * a character. This pointer is then passed and used by the other matrix
83  * routines to refer to a particular matrix. If an error occurs, the NULL
84  * pointer is returned.
85  *
86  * >>> Arguments:
87  * Size <input> (int)
88  * Size of matrix or estimate of size of matrix if matrix is EXPANDABLE.
89  * Complex <input> (int)
90  * Type of matrix. If Complex is 0 then the matrix is real, otherwise

274  * This routine allocates space for matrix elements. It requests large blocks

294 ElementPtr spcGetElement(MatrixPtr Matrix)
295 {
296  ElementPtr pElement;
297 
298  /* Begin `spcGetElement'. */
299 
300  /* Allocate block of MatrixElements if necessary. */
301  if (Matrix->ElementsRemaining == 0) {
302  pElement = ALLOC(struct MatrixElement, ELEMENTS_PER_ALLOCATION);
303  RecordAllocation(Matrix, (char*)pElement);
304  if (Matrix->Error == spNO_MEMORY)
305  return NULL;

307  Matrix->NextAvailElement = pElement;
308  }

315 /*
316  * ELEMENT ALLOCATION INITIALIZATION
317  *
318  * This routine allocates space for matrix fill-ins and an initial set of
319  * elements. Besides being faster than allocating space for elements one
320  * at a time, it tends to keep the fill-ins physically close to the other
321  * matrix elements in the computer memory. This keeps virtual memory paging
322  * to a minimum.
323  *
324  * >>> Arguments:
325  * Matrix <input> (MatrixPtr)
326  * Pointer to the matrix.
327  * InitialNumberOfElements <input> (int)
328  * This number is used as the size of the block of memory, in
329  * MatrixElements, reserved for elements. If more than this number of
330  * elements are generated, then more space is allocated later.
331  * NumberOfFillinsExpected <input> (int)
332  * This number is used as the size of the block of memory, in
333  * MatrixElements, reserved for fill-ins. If more than this number of
334  * fill-ins are generated, then more space is allocated, but they may
335  * not be physically close in computer's memory.
336  *
337  * >>> Local variables:
338  * pElement (ElementPtr)
339  * A pointer to the first element in the group of elements being allocated.
340  *
341  * >>> Possible errors:
342  * spNO_MEMORY
343  */
344 
345 static void InitializeElementBlocks(MatrixPtr Matrix, int InitialNumberOfElements, int NumberOfFillinsExpected)
346 {
347  ElementPtr pElement;
348 
349  /* Begin `InitializeElementBlocks'. */
350 
351  /* Allocate block of MatrixElements for elements. */
352  pElement = ALLOC(struct MatrixElement, InitialNumberOfElements);
353  RecordAllocation(Matrix, (char*)pElement);
354  if (Matrix->Error == spNO_MEMORY)
355  return;
356  Matrix->ElementsRemaining = InitialNumberOfElements;
357  Matrix->NextAvailElement = pElement;
358 
359  /* Allocate block of MatrixElements for fill-ins. */
360  pElement = ALLOC(struct MatrixElement, NumberOfFillinsExpected);
361  RecordAllocation(Matrix, (char*)pElement);
362  if (Matrix->Error == spNO_MEMORY)
363  return;
364  Matrix->FillinsRemaining = NumberOfFillinsExpected;
365  Matrix->NextAvailFillin = pElement;
366 
367  /* Allocate a fill-in list structure. */
368  Matrix->FirstFillinListNode = ALLOC(struct FillinListNodeStruct, 1);
369  RecordAllocation(Matrix, (char*)Matrix->FirstFillinListNode);
370  if (Matrix->Error == spNO_MEMORY)
371  return;
372  Matrix->LastFillinListNode = Matrix->FirstFillinListNode;
373 
374  Matrix->FirstFillinListNode->pFillinList = pElement;
375  Matrix->FirstFillinListNode->NumberOfFillinsInList = NumberOfFillinsExpected;
376  Matrix->FirstFillinListNode->Next = NULL;
377 
378  return;
379 }

382  * FILL-IN ALLOCATION
383  *
384  * This routine allocates space for matrix fill-ins. It requests large blocks
385  * of storage from the system and doles out individual elements as required.
386  * This technique, as opposed to allocating elements individually, tends to
387  * speed the allocation process.
388  *
389  * >>> Returned:
390  * A pointer to the fill-in.
391  *
392  * >>> Arguments:
393  * Matrix <input> (MatrixPtr)
394  * Pointer to matrix.
395  *
396  * >>> Possible errors:
397  * spNO_MEMORY
398  */
399 
400 ElementPtr spcGetFillin(MatrixPtr Matrix)
401 {
402  struct FillinListNodeStruct* pListNode;
403  ElementPtr pFillins;
404 
405  /* Begin `spcGetFillin'. */
406 
407 #if NOT STRIP OR LINT
408  if (Matrix->FillinsRemaining == 0)
409  return spcGetElement(Matrix);
410 #endif
411 #if STRIP OR LINT
412 
413  if (Matrix->FillinsRemaining == 0) {
414  pListNode = Matrix->LastFillinListNode;
415 
416  /* First see if there are any stripped fill-ins left. */
417  if (pListNode->Next != NULL) {
418  Matrix->LastFillinListNode = pListNode = pListNode->Next;
419  Matrix->FillinsRemaining = pListNode->NumberOfFillinsInList;
420  Matrix->NextAvailFillin = pListNode->pFillinList;
421  } else {
422  /* Allocate block of fill-ins. */
423  pFillins = ALLOC(struct MatrixElement, ELEMENTS_PER_ALLOCATION);
424  RecordAllocation(Matrix, (char*)pFillins);
425  if (Matrix->Error == spNO_MEMORY)
426  return NULL;
427  Matrix->FillinsRemaining = ELEMENTS_PER_ALLOCATION;
428  Matrix->NextAvailFillin = pFillins;
429 
430  /* Allocate a fill-in list structure. */
431  pListNode->Next = ALLOC(struct FillinListNodeStruct, 1);
432  RecordAllocation(Matrix, (char*)pListNode->Next);
433  if (Matrix->Error == spNO_MEMORY)
434  return NULL;
435  Matrix->LastFillinListNode = pListNode = pListNode->Next;
436 
437  pListNode->pFillinList = pFillins;
438  pListNode->NumberOfFillinsInList = ELEMENTS_PER_ALLOCATION;
439  pListNode->Next = NULL;
440  }
441  }
442 #endif
443 
444  /* Update Fill-in counter and return pointer to Fill-in. */
445  Matrix->FillinsRemaining--;
446  return Matrix->NextAvailFillin++;
447 }

449 /*
450  * RECORD A MEMORY ALLOCATION
451  *
452  * This routine is used to record all memory allocations so that the memory
453  * can be freed later.
454  *
455  * >>> Arguments:
456  * Matrix <input> (MatrixPtr)
457  * Pointer to the matrix.
458  * AllocatedPtr <input> (char *)
459  * The pointer returned by malloc or calloc. These pointers are saved in
460  * a list so that they can be easily freed.
461  *
462  * >>> Possible errors:
463  * spNO_MEMORY
464  */
465 
466 static void RecordAllocation(MatrixPtr Matrix, char* AllocatedPtr)
467 {
468  /* Begin `RecordAllocation'. */
469  /*
470  * If Allocated pointer is NULL, assume that malloc returned a NULL pointer,
471  * which indicates a spNO_MEMORY error.
472  */
473  if (AllocatedPtr == NULL) {
474  Matrix->Error = spNO_MEMORY;
475  return;
476  }
477 
478  /* Allocate block of MatrixElements if necessary. */
479  if (Matrix->RecordsRemaining == 0) {
480  AllocateBlockOfAllocationList(Matrix);
481  if (Matrix->Error == spNO_MEMORY) {
482  FREE(AllocatedPtr);
483  return;
484  }
485  }
486 
487  /* Add Allocated pointer to Allocation List. */
488  (++Matrix->TopOfAllocationList)->AllocatedPtr = AllocatedPtr;
489  Matrix->RecordsRemaining--;
490  return;
491 }

493 /*
494  * ADD A BLOCK OF SLOTS TO ALLOCATION LIST
495  *
496  * This routine increases the size of the allocation list.
497  *
498  * >>> Arguments:
499  * Matrix <input> (MatrixPtr)
500  * Pointer to the matrix.
501  *
502  * >>> Local variables:
503  * ListPtr (AllocationListPtr)
504  * Pointer to the list that contains the pointers to segments of memory
505  * that were allocated by the operating system for the current matrix.
506  *
507  * >>> Possible errors:
508  * spNO_MEMORY
509  */
510 
511 static void AllocateBlockOfAllocationList(MatrixPtr Matrix)
512 {
513  int I;
514  AllocationListPtr ListPtr;
515 
516  /* Begin `AllocateBlockOfAllocationList'. */
517  /* Allocate block of records for allocation list. */
518  ListPtr = ALLOC(struct AllocationRecord, (ELEMENTS_PER_ALLOCATION + 1));
519  if (ListPtr == NULL) {
520  Matrix->Error = spNO_MEMORY;
521  return;
522  }
523 
524  /* String entries of allocation list into singly linked list. List is linked
525  such that any record points to the one before it. */
526 
527  ListPtr->NextRecord = Matrix->TopOfAllocationList;
528  Matrix->TopOfAllocationList = ListPtr;
529  ListPtr += ELEMENTS_PER_ALLOCATION;
530  for (I = ELEMENTS_PER_ALLOCATION; I > 0; I--) {
531  ListPtr->NextRecord = ListPtr - 1;
532  ListPtr--;
533  }
534 
535  /* Record allocation of space for allocation list on allocation list. */
536  Matrix->TopOfAllocationList->AllocatedPtr = (char*)ListPtr;
537  Matrix->RecordsRemaining = ELEMENTS_PER_ALLOCATION;
538 
539  return;
540 }

542 /*
543  * MATRIX DEALLOCATION
544  *
545  * Deallocates pointers and elements of Matrix.
546  *
547  * >>> Arguments:
548  * Matrix <input> (char *)
549  * Pointer to the matrix frame which is to be removed from memory.
550  *
551  * >>> Local variables:
552  * ListPtr (AllocationListPtr)
553  * Pointer into the linked list of pointers to allocated data structures.
554  * Points to pointer to structure to be freed.
555  * NextListPtr (AllocationListPtr)
556  * Pointer into the linked list of pointers to allocated data structures.
557  * Points to the next pointer to structure to be freed. This is needed
558  * because the data structure to be freed could include the current node
559  * in the allocation list.
560  */
561 
562 void spDestroy(char* eMatrix)
563 {
564  MatrixPtr Matrix = (MatrixPtr)eMatrix;
565  AllocationListPtr ListPtr, NextListPtr;
566 
567  /* Begin `spDestroy'. */
568  ASSERT(IS_SPARSE(Matrix));
569 
570  /* Deallocate the vectors that are located in the matrix frame. */
571  FREE(Matrix->IntToExtColMap);
572  FREE(Matrix->IntToExtRowMap);
573  FREE(Matrix->ExtToIntColMap);
574  FREE(Matrix->ExtToIntRowMap);
575  FREE(Matrix->Diag);
576  FREE(Matrix->FirstInRow);
577  FREE(Matrix->FirstInCol);
578  FREE(Matrix->MarkowitzRow);
579  FREE(Matrix->MarkowitzCol);
580  FREE(Matrix->MarkowitzProd);
581  FREE(Matrix->DoCmplxDirect);
582  FREE(Matrix->DoRealDirect);
583  FREE(Matrix->Intermediate);
584 
585  /* Sequentially step through the list of allocated pointers freeing pointers
586  * along the way. */
587  ListPtr = Matrix->TopOfAllocationList;
588  while (ListPtr != NULL) {
589  NextListPtr = ListPtr->NextRecord;
590  FREE(ListPtr->AllocatedPtr);
591  ListPtr = NextListPtr;
592  }
593  return;
594 }
