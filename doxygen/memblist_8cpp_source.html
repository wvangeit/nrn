56 [[nodiscard]] std::ptrdiff_t Memb_list::legacy_index(double const* ptr) const {
57  assert(m_storage_offset != neuron::container::invalid_row);
58  // For a mechanism with (in order) range variables: a, b[2], c the mechanism data are
59  // ______________________
60  // instance 0 | a b[0] b[1] c |
61  // instance 1 | a' b'[0] b'[1] c' |
62  // instance 2 | a'' b''[0] b''[1] c'' |
63  //
64  // the old layout arranged this as:
65  // [a b[0], b[1], c, a', b'[0], b'[1], c', a'', b''[0], b''[1], c'']
66  // whereas the new layout has three different storage vectors:
67  // [a, a', a'']
68  // [b[0], b[1], b'[0], b'[1], b''[0], b''[1]]
69  // [c, c', c'']
70  // this method, given a pointer into one of the new layout vectors,
71  // calculates the (hypothetical) index into the old (single) vector
72  using Tag = neuron::container::Mechanism::field::FloatingPoint;
73  auto const size = m_storage->size(); // number of instances; 3 in the example above
74  auto const num_fields = m_storage->get_tag<Tag>().num_variables(); // ex: 3 (a, b, c)
75  auto const* const array_dims = m_storage->get_array_dims<Tag>(); // ex: [1, 2, 1]
76  auto const sum_of_array_dims = std::accumulate(array_dims, array_dims + num_fields, 0);
77  int sum_of_array_dims_of_previous_fields{};
78  for (auto field = 0; field < num_fields; ++field) { // a, b or c in the example above
79  auto const array_dim = array_dims[field];
80  assert(array_dim > 0);
81  auto const* const vec_data = &m_storage->get_field_instance<Tag>(0, field);
82  auto const index = std::distance(vec_data, ptr);
83  // storage vectors are size * array_dim long
84  if (index >= 0 && index < size * array_dim) {
85  auto const instance_offset = index / array_dim;
86  auto const array_index = index % array_dim;
87  assert(ptr == &m_storage->get_field_instance<Tag>(instance_offset, field, array_index));
88  return ((instance_offset - m_storage_offset) * sum_of_array_dims) +
89  sum_of_array_dims_of_previous_fields + array_index;
90  }
91  sum_of_array_dims_of_previous_fields += array_dim;
92  }
93  assert(sum_of_array_dims_of_previous_fields == sum_of_array_dims);
94  // ptr doesn't live in this mechanism data, cannot compute a legacy index
95  return -1;
96 }
