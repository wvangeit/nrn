348 void nrn_rhs_ext(NrnThread* _nt) {
349  int i, j, cnt;
350  Node *nd, *pnd, **ndlist;
351  Extnode *nde, *pnde;
352  Memb_list* ml = _nt->_ecell_memb_list;
353  if (!ml) {
354  return;
355  }
356  cnt = ml->nodecount;
357  ndlist = ml->nodelist;
358 
359  /* nd rhs contains -membrane current + stim current */
360  /* nde rhs contains stim current */
361  for (i = 0; i < cnt; ++i) {
362  nd = ndlist[i];
363  nde = nd->extnode;
364  *nde->_rhs[0] -= NODERHS(nd);
365 #if I_MEMBRANE
366  ml->data(i, sav_rhs_index) = *nde->_rhs[0];
367  /* and for daspk this is the ionic current which can be
368  combined later with i_cap before return from solve. */
369 #endif
370  }
371  for (i = 0; i < cnt; ++i) {
372  nd = ndlist[i];
373  nde = nd->extnode;
374  pnd = _nt->_v_parent[nd->v_node_index];
375  if (pnd) {
376  pnde = pnd->extnode;
377  /* axial contributions */
378  if (pnde) { /* parent sec may not be extracellular */
379  for (j = 0; j < nrn_nlayer_extracellular; ++j) {
380  double dv = pnde->v[j] - nde->v[j];
381  *nde->_rhs[j] -= nde->_b[j] * dv;
382  *pnde->_rhs[j] += nde->_a[j] * dv;
383  /* for the internal balance equation
384  take care of vi = vm + vx */
385  if (j == 0) {
386  NODERHS(nd) -= NODEB(nd) * dv;
387  NODERHS(pnd) += NODEA(nd) * dv;
388  }
389  }
390  } else { /* no extracellular in parent but still need
391  to take care of vi = vm + vx in this node.
392  Note that we are NOT grounding the parent
393  side of xraxial so equivalent to sealed.
394  Also note that when children of this node
395  do not have extracellular, we deal with
396  that, at end of this function. */
397  double dv = -nde->v[0];
398  NODERHS(nd) -= NODEB(nd) * dv;
399  NODERHS(pnd) += NODEA(nd) * dv;
400  }
401 
402  /* series resistance and battery to ground */
403  /* between nlayer-1 and ground */
404  j = nrn_nlayer_extracellular - 1;
405  *nde->_rhs[j] -= *nde->param[xg_index_ext(j)] *
406  (nde->v[j] - *nde->param[e_extracellular_index_ext()]);
407  for (--j; j >= 0; --j) { /* between j and j+1 layer */
408  double x = *nde->param[xg_index_ext(j)] * (nde->v[j] - nde->v[j + 1]);
409  *nde->_rhs[j] -= x;
410  *nde->_rhs[j + 1] += x;
411  }
412  }
413  }
414  cnt = _nt->_ecell_child_cnt;
415  for (i = 0; i < cnt; ++i) {
416  double dv;
417  nd = _nt->_ecell_children[i];
418  pnd = _nt->_v_parent[nd->v_node_index];
419  dv = pnd->extnode->v[0];
420  NODERHS(nd) -= NODEB(nd) * dv;
421  NODERHS(pnd) += NODEA(nd) * dv;
422  }
423 }

425 void nrn_setup_ext(NrnThread* _nt) {
426  int i, j, cnt;
427  Node *nd, *pnd, **ndlist;
428  double* pd;
429  double d, cfac, mfac;
430  Extnode *nde, *pnde;
431  Memb_list* ml = _nt->_ecell_memb_list;
432  if (!ml) {
433  return;
434  }
435  /*printnode("begin setup");*/
436  cnt = ml->nodecount;
437  ndlist = ml->nodelist;
438  cfac = .001 * _nt->cj;
439 
440  /* d contains all the membrane conductances (and capacitance) */
441  /* i.e. (cm/dt + di/dvm - dis/dvi)*[dvi] and
442  (dis/dvi)*[dvx] */
443  for (i = 0; i < cnt; ++i) {
444  nd = ndlist[i];
445  nde = nd->extnode;
446  d = NODED(nd);
447  /* nde->_d only has -ELECTRODE_CURRENT contribution */
448  d = (*nde->_d[0] += NODED(nd));
449  /* now d is only the membrane current contribution */
450  /* i.e. d = cm/dt + di/dvm */
451  *nde->_x12[0] -= d;
452  *nde->_x21[0] -= d;
453 #if I_MEMBRANE
454  ml->data(i, sav_g_index) = d;
455 #endif
456  }
457  /* series resistance, capacitance, and axial terms. */
458  for (i = 0; i < cnt; ++i) {
459  nd = ndlist[i];
460  nde = nd->extnode;
461  pnd = _nt->_v_parent[nd->v_node_index];
462  if (pnd) {
463  /* series resistance and capacitance to ground */
464  j = 0;
465  for (;;) { /* between j and j+1 layer */
466  mfac = (*nde->param[xg_index_ext(j)] + *nde->param[xc_index_ext(j)] * cfac);
467  *nde->_d[j] += mfac;
468  ++j;
469  if (j == nrn_nlayer_extracellular) {
470  break;
471  }
472  *nde->_d[j] += mfac;
473  *nde->_x12[j] -= mfac;
474  *nde->_x21[j] -= mfac;
475  }
476  pnde = pnd->extnode;
477  /* axial connections */
478  if (pnde) { /* parent sec may not be extracellular */
479  for (j = 0; j < nrn_nlayer_extracellular; ++j) {
480  *nde->_d[j] -= nde->_b[j];
481  *pnde->_d[j] -= nde->_a[j];
482  *nde->_a_matelm[j] += nde->_a[j];
483  *nde->_b_matelm[j] += nde->_b[j];
484  }
485  }
486  }
487  }
488  /*printnode("end setup_lhs");*/
489 }

492 void ext_con_coef(void) /* setup a and b */
493 {

495  double dx, area;

497  Node *nd, **pnd;
498  Extnode* nde;

501  /* temporarily store half segment resistances in rhs */
502  // ForAllSections(sec)
503  ITERATE(qsec, section_list) {
504  Section* sec = hocSEC(qsec);
505  if (sec->pnode[0]->extnode) {
506  dx = section_length(sec) / ((double) (sec->nnode - 1));
507  for (j = 0; j < sec->nnode - 1; j++) {
508  nde = sec->pnode[j]->extnode;
509  for (k = 0; k < nrn_nlayer_extracellular; ++k) {

511  (dx / 2.); /*Megohms*/
512  }
513  }
514  /* last segment has 0 length. */
515  nde = sec->pnode[j]->extnode;
516  for (k = 0; k < nrn_nlayer_extracellular; ++k) {
517  *nde->_rhs[k] = 0.;
518  *nde->param[xc_index_ext(k)] = 0.;
519  *nde->param[xg_index_ext(k)] = 0.;
520  }
521  /* if owns a rootnode */
522  if (!sec->parentsec) {
523  nde = sec->parentnode->extnode;
524  for (k = 0; k < nrn_nlayer_extracellular; ++k) {
525  *nde->_rhs[k] = 0.;
526  *nde->param[xc_index_ext(k)] = 0.;
527  *nde->param[xg_index_ext(k)] = 0.;
528  }
529  }
530  }
531  }
532  /* assume that if only one connection at x=1, then they butte
533  together, if several connections at x=1
534  then last point is at x=1, has 0 area and other points are at
535  centers of nnode-1 segments.
536  If interior connection then child half
537  section connects straight to the point*/
538  /* for the near future we always have a last node at x=1 with
539  no properties */
540  // ForAllSections(sec)
541  ITERATE(qsec, section_list) {
542  Section* sec = hocSEC(qsec);
543  if (sec->pnode[0]->extnode) {
544  /* node half resistances in general get added to the
545  node and to the node's "child node in the same section".
546  child nodes in different sections don't involve parent
547  node's resistance */
548  nde = sec->pnode[0]->extnode;
549  for (k = 0; k < nlayer; ++k) {
550  nde->_b[k] = *nde->_rhs[k];
551  }
552  for (j = 1; j < sec->nnode; j++) {
553  nde = sec->pnode[j]->extnode;
554  for (k = 0; k < nlayer; ++k) {
555  nde->_b[k] = *nde->_rhs[k] + *(sec->pnode[j - 1]->extnode->_rhs[k]); /*megohms*/
556  }
557  }
558  }
559  }
560  // ForAllSections(sec)
561  ITERATE(qsec, section_list) {
562  Section* sec = hocSEC(qsec);
563  if (sec->pnode[0]->extnode) {
564  /* convert to siemens/cm^2 for all nodes except last
565  and microsiemens for last. This means that a*V = mamps/cm2
566  and a*v in last node = nanoamps. Note that last node
567  has no membrane properties and no area. It may perhaps recieve
568  current stimulus later */
569  /* first the effect of node on parent equation. Note That
570  last nodes have area = 1.e2 in dimensionless units so that
571  last nodes have units of microsiemens's */
572  pnd = sec->pnode;
573  nde = pnd[0]->extnode;
574  area = NODEAREA(sec->parentnode);
575  /* param[4] is rall_branch */
576  for (k = 0; k < nlayer; ++k) {
577  nde->_a[k] = -1.e2 * sec->prop->dparam[4].get<double>() / (nde->_b[k] * area);
578  }
579  for (j = 1; j < sec->nnode; j++) {
580  nde = pnd[j]->extnode;
581  area = NODEAREA(pnd[j - 1]);
582  for (k = 0; k < nlayer; ++k) {
583  nde->_a[k] = -1.e2 / (nde->_b[k] * area);
584  }
585  }
586  }
587  }
588  /* now the effect of parent on node equation. */
589  // ForAllSections(sec)
590  ITERATE(qsec, section_list) {
591  Section* sec = hocSEC(qsec);
