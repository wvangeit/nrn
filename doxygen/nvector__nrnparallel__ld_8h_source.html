94 /*
95  * -----------------------------------------------------------------
96  * PART II: macros NV_CONTENT_P_LD, NV_DATA_P_LD, NV_OWN_DATA_P_LD,
97  * NV_LOCLENGTH_P_LD, NV_GLOBLENGTH_P_LD,NV_COMM_P_LD, and NV_Ith_P_LD
98  * -----------------------------------------------------------------
99  * In the descriptions below, the following user declarations
100  * are assumed:
101  *
102  * N_Vector v;
103  * long int v_len, s_len, i;
104  *
105  * (1) NV_CONTENT_P_LD
106  *
107  * This routines gives access to the contents of the parallel
108  * vector N_Vector.
109  *
110  * The assignment v_cont = NV_CONTENT_P_LD(v) sets v_cont to be
111  * a pointer to the parallel N_Vector content structure.
112  *
113  * (2) NV_DATA_P_LD, NV_OWN_DATA_P_LD, NV_LOCLENGTH_P_LD, NV_GLOBLENGTH_P_LD,
114  * and NV_COMM_P_LD
115  *
116  * These routines give access to the individual parts of
117  * the content structure of a parallel N_Vector.
118  *
119  * The assignment v_data = NV_DATA_P_LD(v) sets v_data to be
120  * a pointer to the first component of the local data for
121  * the vector v. The assignment NV_DATA_P_LD(v) = data_v sets
122  * the component array of v to be data_V by storing the
123  * pointer data_v.
124  *
125  * The assignment v_llen = NV_LOCLENGTH_P_LD(v) sets v_llen to
126  * be the length of the local part of the vector v. The call
127  * NV_LOCLENGTH_P_LD(v) = llen_v sets the local length
128  * of v to be llen_v.
129  *
130  * The assignment v_glen = NV_GLOBLENGTH_P_LD(v) sets v_glen to
131  * be the global length of the vector v. The call
132  * NV_GLOBLENGTH_P_LD(v) = glen_v sets the global length of v to
133  * be glen_v.
134  *
135  * The assignment v_comm = NV_COMM_P_LD(v) sets v_comm to be the
136  * MPI communicator of the vector v. The assignment
137  * NV_COMM_C(v) = comm_v sets the MPI communicator of v to be
138  * comm_v.
139  *
140  * (3) NV_Ith_P_LD
141  *
142  * In the following description, the components of the
143  * local part of an N_Vector are numbered 0..n-1, where n
144  * is the local length of (the local part of) v.
145  *
146  * The assignment r = NV_Ith_P_LD(v,i) sets r to be the value
147  * of the ith component of the local part of the vector v.
148  * The assignment NV_Ith_P_LD(v,i) = r sets the value of the
149  * ith local component of v to be r.
150  *
151  * Note: When looping over the components of an N_Vector v, it is
152  * more efficient to first obtain the component array via
153  * v_data = NV_DATA_P_LD(v) and then access v_data[i] within the
154  * loop than it is to use NV_Ith_P_LD(v,i) within the loop.
155  * -----------------------------------------------------------------
156  */
157 
158 #define NV_CONTENT_P_LD(v) ((N_VectorContent_NrnParallelLD) (v->content))

162 #define NV_GLOBLENGTH_P_LD(v) (NV_CONTENT_P_LD(v)->global_length)
163 
164 #define NV_OWN_DATA_P_LD(v) (NV_CONTENT_P_LD(v)->own_data)
165 
166 #define NV_DATA_P_LD(v) (NV_CONTENT_P_LD(v)->data)

168 #define NV_COMM_P_LD(v) (NV_CONTENT_P_LD(v)->comm)
