302 void whileloop(Item* qsol, long type, int ss) {
303  /* no solve statement except this is allowed to
304  change the indepvar. Time passed to integrators
305  must therefore be saved and restored. */
306  /* Note that when the user changes the independent variable within Scop,
307  the while loop still uses the original independent variable. In this case
308  1) _break must be set to the entry value of the original independent variable
309  2) initmodel() must be called after each entry to model()
310  3) in initmodel() we are very careful not to destroy the entry value of
311  time-- on exit it has its original value. Note that _sav_indep gets
312  any value of time that is set within initmodel().
313  */
314  /* executing more that one for loop in a single call to model() is an error
315  which is trapped in scop */
316  static int called = 0, firstderf = 1;
317  const char* cp = 0;
318 
319  switch (type) {
320  case DERF:
321  case DISCRETE:
322  if (firstderf) { /* time dependent process so create a deltastep */
323  double d[3];
324  int i;
325  Item* q;
326  char sval[256];
327  Sprintf(buf, "delta_%s", indepsym->name);
328  for (i = 0, q = indeplist->next; i < 3; i++, q = q->next) {
329  d[i] = atof(STR(q));
330  }
331  if (type == DERF) {
332  Sprintf(sval, "%g", (d[1] - d[0]) / d[2]);
333  } else if (type == DISCRETE) {
334  Sprintf(sval, "1.0");
335  }
336  deltaindep = ifnew_parminstall(buf, sval, "", "");
337  firstderf = 0;
338  }
339  if (type == DERF) {
340  cp = "dt";
341  } else if (type == DISCRETE) {
342  cp = "0.0";
343  }
344  if (ss) {
345  return;
346  }
347  break;
348  default:
349  /*SUPPRESS 622*/
350  assert(0);
351  }
352 
353  if (called) {
354  Fprintf(stderr,
355  "Warning: More than one integrating SOLVE statement in an \
356 BREAKPOINT block.\nThe simulation will be incorrect if more than one is used \
357 at a time.\n");
358  }
359  if (!called) {
360  /* fix up initmodel as per 3) above.
361  In cout.c _save is declared */
362  Sprintf(buf, " _save = %s;\n %s = 0.0;\n", indepsym->name, indepsym->name);
363  saveindep = stralloc(buf, (char*) 0);
364  /* Assert no more additions to initfunc involving
365  the value of time */
366  Sprintf(buf, " _sav_indep = %s; %s = _save;\n", indepsym->name, indepsym->name);
367  Lappendstr(initfunc, buf);
