53 /** @brief permuted array copied to unpermuted array
54  * If permute is NULL then just a copy
55  */
56 static void inverse_permute_copy(size_t n, double* permuted_src, double* dest, int* permute) {
57  if (permute) {
58  for (size_t i = 0; i < n; ++i) {
59  dest[i] = permuted_src[permute[i]];
60  }
61  } else {
62  std::copy(permuted_src, permuted_src + n, dest);
63  }
64 }

66 /** @brief SoA permuted mechanism data copied to unpermuted AoS data.
67  * dest is an array of n pointers to the beginning of each sz length array.
68  * src is a contiguous array of sz segments of size stride. The stride
69  * may be slightly greater than n for purposes of alignment.
70  * Each of the sz segments of src are permuted.
71  */
72 static void soa2aos_inverse_permute_copy(size_t n,
73  int sz,
74  int stride,
75  double* src,
76  std::vector<double*>& dest,
77  int* permute) {
78  // src is soa and permuted. dest is n pointers to sz doubles (aos).
79  for (size_t instance = 0; instance < n; ++instance) {
80  double* s = src + permute[instance];
81  for (int i = 0; i < sz; ++i) {
82  dest[i][instance] = s[i * stride];
83  }
84  }
85 }

87 /** @brief SoA unpermuted mechanism data copied to unpermuted AoS data.
88  * dest is an array of n pointers to the beginning of each sz length array.
89  * src is a contiguous array of sz segments of size stride. The stride
90  * may be slightly greater than n for purposes of alignment.
91  * Each of the sz segments of src have the same order as the n pointers
92  * of dest.
93  */
94 static void soa2aos_unpermuted_copy(size_t n,
95  int sz,
96  int stride,
97  double* src,
98  std::vector<double*>& dest) {
99  // src is soa and permuted. dest is n pointers to sz doubles (aos).
100  for (size_t instance = 0; instance < n; ++instance) {
101  double* s = src + instance;
102  for (int i = 0; i < sz; ++i) {
103  dest[i][instance] = s[i * stride];
104  }
105  }
106 }

108 /** @brief AoS mechanism data copied to AoS data.
109  * dest is an array of n pointers to the beginning of each sz length array.
110  * src is a contiguous array of n segments of size sz.
111  */
112 static void aos2aos_copy(size_t n, int sz, double* src, std::vector<double*>& dest) {
113  for (size_t instance = 0; instance < n; ++instance) {
114  double* s = src + (instance * sz);
115  for (auto i = 0; i < sz; ++i) {
116  dest[i][instance] = s[i];
117  }
118  }

123 static void core2nrn_corepointer(int tid, NrnThreadMembList* tml) {
124  // Based on get_bbcore_write fragment in nrn_checkpoint.cpp
125  int type = tml->index;
126  if (!corenrn.get_bbcore_write()[type]) {
127  return;
128  }
129  NrnThread& nt = nrn_threads[tid];
130  Memb_list* ml = tml->ml;
131  double* d = nullptr;
132  Datum* pd = nullptr;
133  int layout = corenrn.get_mech_data_layout()[type];
134  int dsz = corenrn.get_prop_param_size()[type];
135  int pdsz = corenrn.get_prop_dparam_size()[type];
136  int aln_cntml = nrn_soa_padded_size(ml->nodecount, layout);
137 
138  int icnt = 0;
139  int dcnt = 0;
140  // data size and allocate
141  for (int j = 0; j < ml->nodecount; ++j) {
142  int jp = j;
143  if (ml->_permute) {
144  jp = ml->_permute[j];
145  }
146  d = ml->data + nrn_i_layout(jp, ml->nodecount, 0, dsz, layout);
147  pd = ml->pdata + nrn_i_layout(jp, ml->nodecount, 0, pdsz, layout);
148  (*corenrn.get_bbcore_write()[type])(
149  nullptr, nullptr, &dcnt, &icnt, 0, aln_cntml, d, pd, ml->_thread, &nt, ml, 0.0);
150  }
151 
152  std::unique_ptr<int[]> iArray;
153  std::unique_ptr<double[]> dArray;
154  if (icnt) {
155  iArray.reset(new int[icnt]);
156  }
157  if (dcnt) {
158  dArray.reset(new double[dcnt]);
159  }
160  icnt = dcnt = 0;
161  for (int j = 0; j < ml->nodecount; j++) {
162  int jp = j;
163 
164  if (ml->_permute) {
165  jp = ml->_permute[j];
166  }
167 
168  d = ml->data + nrn_i_layout(jp, ml->nodecount, 0, dsz, layout);
169  pd = ml->pdata + nrn_i_layout(jp, ml->nodecount, 0, pdsz, layout);
170 
171  (*corenrn.get_bbcore_write()[type])(dArray.get(),
172  iArray.get(),
173  &dcnt,
174  &icnt,
175  0,
176  aln_cntml,
177  d,
178  pd,
179  ml->_thread,
180  &nt,
181  ml,
182  0.0);
183  }

235  nrnran123_State* state = (nrnran123_State*) nt._vdata[pv];
236  uint32_t seq;
237  char which;
238  nrnran123_getseq(state, &seq, &which);
239  nmodlrandom.push_back(double(seq) * 4 + which);
240  }

280 /** @brief copy data back to NEURON.
281  * Copies t, voltage, i_membrane_ if it used, and mechanism param data.
282  * Copies event queue and related state, e.g. WATCH, VecPlayContinuous.
283  */
284 void core2nrn_data_return() {
285  if (!nrn2core_type_return_) {
286  return;
287  }
288 
289  (*core2nrn_clear_queues_)(nrn_threads[0]._t); // all threads at same time
290 
291  for (int tid = 0; tid < nrn_nthread; ++tid) {
292  size_t n = 0;
293  double* data = nullptr;
294  NrnThread& nt = nrn_threads[tid];
295  std::vector<double*> mdata{};
296  n = (*nrn2core_type_return_)(0, tid, data, mdata); // 0 means time
297  if (n) { // not the empty thread
298  data[0] = nt._t;
299  }
300 
301  if (nt.end) { // transfer voltage and possibly i_membrane_
302  n = (*nrn2core_type_return_)(voltage, tid, data, mdata);
303  assert(n == size_t(nt.end) && data);
304  inverse_permute_copy(n, nt._actual_v, data, nt._permute);
305 
306  if (nt.nrn_fast_imem) {
307  n = (*nrn2core_type_return_)(i_membrane_, tid, data, mdata);
308  assert(n == size_t(nt.end) && data);
309  inverse_permute_copy(n, nt.nrn_fast_imem->nrn_sav_rhs, data, nt._permute);
310  }
311  }
312 
313  for (NrnThreadMembList* tml = nt.tml; tml; tml = tml->next) {
314  int mtype = tml->index;
315  Memb_list* ml = tml->ml;
316  n = (*nrn2core_type_return_)(mtype, tid, data, mdata);
317  assert(n == size_t(ml->nodecount) && !mdata.empty());
318  if (n == 0) {
319  continue;
320  }
321  // NEURON is AoS, CoreNEURON may be SoA and may be permuted.
322  // On the NEURON side, the data is actually contiguous because of
323  // cache_efficient, but that may not be the case for ARTIFICIAL_CELL.
324  // For initial implementation simplicity, use the mdata info which gives
325  // a double* for each param_size mech instance.
326  int* permute = ml->_permute;
327  double* cndat = ml->data;
328  int layout = corenrn.get_mech_data_layout()[mtype];
329  int sz = corenrn.get_prop_param_size()[mtype];
330  if (layout == Layout::SoA) {
331  int stride = ml->_nodecount_padded;
332  if (permute) {
333  soa2aos_inverse_permute_copy(n, sz, stride, cndat, mdata, permute);
334  } else {
335  soa2aos_unpermuted_copy(n, sz, stride, cndat, mdata);
336  }
337  } else { /* AoS */
338  aos2aos_copy(n, sz, cndat, mdata);
339  }
340 
341  core2nrn_corepointer(tid, tml);
342  c2n_nmodlrandom(tid, tml);
343  }
344 
345  // Copy the event queue and related state.
346  core2nrn_tqueue(nt);
347  }
348  core2nrn_vecplay();
349  core2nrn_watch();
350 }

352 /** @brief Callbacks into NEURON for WatchCondition.
353  */
354 static void core2nrn_watch() {
355  (*core2nrn_watch_clear_)();
356 
357  // much of the following nested iterations follows the
358  // watch_activate_clear() function in sim/finitialize.cpp, though here
359  // we iterate over nt._watch_types instead of nt.tml and then picking out
360  // the WATCH relevant types with corenrn.get_watch_check().
361  for (int tid = 0; tid < nrn_nthread; ++tid) {
362  NrnThread& nt = nrn_threads[tid];
363  if (nt._watch_types) {
364  for (int i = 0; nt._watch_types[i] != 0; ++i) {
365  int type = nt._watch_types[i];
366  Memb_list& ml = *(nt._ml_list[type]);
367  int nodecount = ml.nodecount;
368  Core2NrnWatchInfo watch_info(ml.nodecount);
369  int* permute = ml._permute;
370  int* pdata = (int*) ml.pdata;
371  int dparam_size = corenrn.get_prop_dparam_size()[type];
372  int layout = corenrn.get_mech_data_layout()[type];
373  int first, last;
374  watch_datum_indices(type, first, last);
375  int watch_begin = first;
376  for (int iml = 0; iml < nodecount; ++iml) {
377  int iml_permute = permute ? permute[iml] : iml;
378  Core2NrnWatchInfoItem& wiv = watch_info[iml];
379  for (int ix = first; ix <= last; ++ix) {
380  int datum =
381  pdata[nrn_i_layout(iml_permute, nodecount, ix, dparam_size, layout)];
382  if (datum & 2) { // activated
383  bool above_thresh = bool(datum & 1);
384  wiv.push_back(std::pair<int, bool>(ix, above_thresh));
385  }
386  }
387  }
388  (*core2nrn_watch_activate_)(tid, type, watch_begin, watch_info);
389  }
390  }
391  }

509 // return false unless q is pushed to sewm
510 static bool core2nrn_tqueue_item(TQItem* q, SelfEventWeightMap& sewm, NrnThread& nt) {
511  DiscreteEvent* d = (DiscreteEvent*) q->data_;
512  double td = q->t_;
513  bool in_sewm = false;
514 
515  switch (d->type()) {
516  case NetConType: {
517  NetCon* nc = (NetCon*) d;
518  assert(nc >= nt.netcons && (nc < (nt.netcons + nt.n_netcon)));
519  size_t nc_index = nc - nt.netcons;
520  (*core2nrn_NetCon_event_)(nt.id, td, nc_index);
521  break;
522  }
523  case SelfEventType: {
524  SelfEvent* se = (SelfEvent*) d;
525  Point_process* pnt = se->target_;
526  assert(pnt->_tid == nt.id);
527  int tar_type = (int) pnt->_type;
528  Memb_list* ml = nt._ml_list[tar_type];
529  if (ml->_permute) { // if permutation, then make inverse available
530  // Doing this here because we don't know, in general, which
531  // mechanisms use SelfEvent
532  if (type2invperm.count(tar_type) == 0) {
533  type2invperm[tar_type] = inverse_permute(ml->_permute, ml->nodecount);
534  }
535  }
536  double flag = se->flag_;
537  TQItem** movable = (TQItem**) (se->movable_);
538  int is_movable = (movable && *movable == q) ? 1 : 0;
539  int weight_index = se->weight_index_;
540  // the weight_index is useless on the NEURON side so we need
541  // to convert that to NetCon index and let the NEURON side
542  // figure out the weight_index. To figure out the netcon_index
543  // construct a {weight_index : [TQItem]} here for any
544  // weight_index >= 0, otherwise send it NEURON now.
545  if (weight_index >= 0) {
546  // Potentially several SelfEvent TQItem* associated with
547  // same weight index. More importantly, collect them all
548  // so that we only need to iterate over the nt.netcons once
549  sewm[weight_index].push_back(q);
550  in_sewm = true;
551 
552  } else {
553  int tar_index = pnt->_i_instance; // correct for no permutation
554  if (ml->_permute) {
555  tar_index = type2invperm[tar_type][tar_index];
556  }
557  (*core2nrn_SelfEvent_event_noweight_)(nt.id, td, tar_type, tar_index, flag, is_movable);
558  delete se;
559  }
560  break;
561  }
562  case PreSynType: {
563  // nothing to transfer
564  // `d` can be cast to PreSyn*
565  break;
566  }
567  case NetParEventType: {
568  // nothing to transfer
569  break;
570  }
571  case PlayRecordEventType: {
572  // nothing to transfer
573  break;
574  }
575  case ReportEventType: {
576  // no need to transfer ReportEvent
577  break;
578  }
579  default: {
580  // In particular, InputPreSyn does not appear in tqueue as it

592 void core2nrn_tqueue(NrnThread& nt) {
593  // VecPlayContinuous
594 
595  // PatternStim
596 
597  // nrn_checkpoint.cpp has:
598  // Avoid extra spikes due to some presyn voltages above threshold
599 
600  // PreSyn.flag_ that are on
601  core2nrn_PreSyn_flag(nt);
602 
603  // The items on the queue
604  NetCvodeThreadData& ntd = net_cvode_instance->p[nt.id];
605  // make sure all buffered interthread events are on the queue
606  ntd.enqueue(net_cvode_instance, &nt);
607 
608  TQueue<QTYPE>* tqe = ntd.tqe_;
609  TQItem* q;
610  SelfEventWeightMap sewm;
611  // TQItems from atomic_dq
612  while ((q = tqe->atomic_dq(1e20)) != nullptr) {
613  if (core2nrn_tqueue_item(q, sewm, nt) == false) {
614  delete q;
615  }
616  }
617  // TQitems from binq_
618  for (q = tqe->binq_->first(); q; q = tqe->binq_->next(q)) {
619  bool const result = core2nrn_tqueue_item(q, sewm, nt);
620  assert(result == false);
621  }
622 
623  // For self events with weight, find the NetCon index and send that
624  // to NEURON.
625  // If the SelfEventWeightMap approach (and the corresponding pattern
626  // on the nrn2core side in NEURON) ends up being too expensive in space
627  // or time, it would be possible to modify SelfEvent to use the NetCon
628  // index instead of the weight index, and then directly determine the
629  // NetCon within the core2nrn_tqueue_item function above and call
630  // (*core2nrn_SelfEvent_event_) from there.
631  if (!sewm.empty()) {
632  for (int nc_index = 0; nc_index < nt.n_netcon; ++nc_index) {
633  NetCon& nc = nt.netcons[nc_index];
634  int weight_index = nc.u.weight_index_;
635  auto search = sewm.find(weight_index);
636  if (search != sewm.end()) {
637  const auto& tqitems = search->second;
638  for (auto q: tqitems) {
639  DiscreteEvent* d = (DiscreteEvent*) (q->data_);
640  double td = q->t_;
641  assert(d->type() == SelfEventType);
642  SelfEvent* se = (SelfEvent*) d;
643  int tar_type = se->target_->_type;
644  // Note that instead of getting tar_index from the permuted
645  // pnt->_i_instance here and for the noweight case above
646  // which then needs the possibly large inverse permutation
647  // vectors, it would save some space to use the unpermuted
648  // nt.pntprocs array along with a much shorter vector
649  // of type offsets.
650  int tar_index = se->target_->_i_instance;
651  if (nt._ml_list[tar_type]->_permute) {
652  tar_index = type2invperm[tar_type][tar_index];
653  }
654  double flag = se->flag_;
655  TQItem** movable = (TQItem**) (se->movable_);
656  int is_movable = (movable && *movable == q) ? 1 : 0;
657  (*core2nrn_SelfEvent_event_)(
658  nt.id, td, tar_type, tar_index, flag, nc_index, is_movable);
659  delete q;
660  delete se;
661  }
662  }
663  }
664  }
665 
666  clear_inv_perm_for_selfevent_targets();
667 }
