45 static int add_arg(const char* name, const char* value) {
46  if (size_t(argc + 2 + bool(value)) >= arg_size) {
47  arg_size += 10;
48  argv = (char**) realloc(argv, arg_size * sizeof(char*));
49  }
50  // Don't add if already in argv
51  for (int i = 1; i < argc; ++i) {
52  if (strcmp(name, argv[i]) == 0) {
53  return 0;
54  }
55  }
56  argv[argc++] = strdup(name);
57  if (value) {
58  argv[argc++] = strdup(value);
59  }
60  // To match C/C++ semantics for the main function then argv[argc] should
61  // always be null. https://en.cppreference.com/w/cpp/language/main_function,
62  // for example, says argv is a "Pointer to the first element of an array of
63  // argc + 1 pointers, of which the last one is null". Section 11.2.1 of the
64  // MPI 4.0 standard says "MPI_INIT accepts the argc and argv that are provided
65  // by the arguments to main or NULL", so it seems correct to follow the rules
66  // for a C/C++ main function. More pragmatically, OpenMPI assumes argv[argc]
67  // is nullptr and crashes if it is not.
68  argv[argc] = NULL;
69  return 1;
70 }

131  int rval = 0;

134  }

136  // int state = 0; // 1 means in "...", 2 means in '...'
137  for (char* cp = s; *cp; cp++) {
138  while (isspace(*cp)) { // skip spaces
139  ++cp;
140  if (*cp == '\0') {
141  free(s);
142  return rval;
143  }
144  }
145  // start processing a token
146  char* cpbegin = cp;
147  char* cp1 = cpbegin; // in token pointer, escapes cause to lag behind
148  while (!isspace(*cp) && *cp != '\0') { // to next space delimiter
149  *cp1++ = *cp++;
150  if (cp1[-1] == '\\' && (isspace(*cp) || *cp == '"' || *cp == '\'')) {
151  // escaped space, ", or '
152  cp1[-1] = *cp++;
153  } else if (cp1[-1] == '"') { // consume to next (unescaped) "
154  cp1--; // backup over the "
155  while (*cp != '"' && *cp != '\0') {
156  *cp1++ = *cp++;
157  if (cp1[-1] == '\\' && *cp == '"') { // escaped " inside "..."
158  cp1[-1] = *cp++;
159  }
160  }
161  if (*cp == '"') {
162  cp++; // skip over the closing "
163  }
164  } else if (cp1[-1] == '\'') { // consume to next (unescaped) '
165  cp1--; // backup over the '
166  while (*cp != '\'' && *cp != '\0') {
167  *cp1++ = *cp++;
168  if (cp1[-1] == '\\' && *cp == '\'') { // escaped ' inside '...'
169  cp1[-1] = *cp++;
170  }
171  }
172  if (*cp == '\'') {
173  cp++; // skip over the closing '
174  }
175  }
176  }
177  if (*cp == '\0') { // at end of s. 'for' will return after it increments.
178  --cp;
179  }
180  if (cp1 > cpbegin) {
181  *cp1 = '\0';
182  }
183  if (strcmp(cpbegin, "-print-options") == 0) {
184  rval = 1;
