80 void reset_sys_path(std::string_view new_first) {
81  PyLockGIL _{};
82  auto* const path = PySys_GetObject("path");
83  nrn_assert(path);
84  // Clear sys.path
85  nrn_assert(PyList_SetSlice(path, 0, PyList_Size(path), nullptr) != -1);
86  // Decode new_first and make a Python unicode string out of it
87  auto* const ustr = PyUnicode_DecodeFSDefaultAndSize(new_first.data(), new_first.size());

89  // Put the decoded string into sys.path
