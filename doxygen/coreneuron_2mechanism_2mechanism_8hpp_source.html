104  void grow() {

120  private:
121  template <typename T>
122  void grow_buf(T** buf, int size, int new_size) {
123  T* new_buf = nullptr;
124  new_buf = (T*) ecalloc_align(new_size, sizeof(T));
125  memcpy(new_buf, *buf, size * sizeof(T));
126  free(*buf);
127  *buf = new_buf;
