110  /** Parse a neuron section segment mapping
111  *
112  * Read count no of mappings for section to segment
113  */
114  template <typename T>
115  int read_mapping_info(T* mapinfo, NrnThreadMappingInfo* ntmapping, CellMapping* cmap) {
116  int nsec, nseg, n_scan;
117  size_t total_lfp_factors;
118  int num_electrodes;
119  char line_buf[max_line_length], name[max_line_length];
120 
121  F.getline(line_buf, sizeof(line_buf));
122  n_scan = sscanf(
123  line_buf, "%s %d %d %zd %d", name, &nsec, &nseg, &total_lfp_factors, &num_electrodes);
124 
125  nrn_assert(n_scan == 5);
126 
127  mapinfo->name = std::string(name);
128 
129  if (nseg) {
130  auto sec = read_vector<int>(nseg);
131  auto seg = read_vector<int>(nseg);
132 
133  std::vector<double> lfp_factors;
134  if (total_lfp_factors > 0) {
135  // ASan reports container overflow on read_array with vec.reserve, resize does work
136  lfp_factors.resize(nseg);
137  read_array<double>(&lfp_factors[0], total_lfp_factors);
138  }
139 
140  int factor_offset = 0;
141  for (int i = 0; i < nseg; i++) {
142  mapinfo->add_segment(sec[i], seg[i]);
143  ntmapping->add_segment_id(seg[i]);
144  int factor_offset = i * num_electrodes;
145  if (total_lfp_factors > 0) {
146  // Abort if the factors contains a NaN
147  nrn_assert(count_if(lfp_factors.begin(), lfp_factors.end(), [](double d) {
148  return std::isnan(d);
149  }) == 0);
150  std::vector<double> segment_factors(lfp_factors.begin() + factor_offset,
151  lfp_factors.begin() + factor_offset +
152  num_electrodes);
153  cmap->add_segment_lfp_factor(seg[i], segment_factors);
154  }
155  }
156  }
157  return nseg;
158  }

160  /** Defined flag values for parse_array() */
161  enum parse_action { read, seek };

163  /** Generic parse function for an array of fixed length.
164  *
165  * \tparam T the array element type: may be \c int or \c double.
166  * \param p pointer to the target in memory for reading into.
167  * \param count number of items of type \a T to parse.
168  * \param action whether to validate and skip (\c seek) or
169  * copy array into memory (\c read).
170  * \return the supplied pointer value.
171  *
172  * Error if \a count is non-zero, \a flag is \c read, and
173  * the supplied pointer \p is null.
174  *
175  * Arrays are represented by a checkpoint line followed by
176  * the array items in increasing index order, in the native binary
177  * representation of the writing process.
178  */
179  template <typename T>
180  inline T* parse_array(T* p, size_t count, parse_action flag) {
181  if (count > 0 && flag != seek)
182  nrn_assert(p != 0);
183 
184  read_checkpoint_assert();
185  switch (flag) {
186  case seek:
187  F.seekg(count * sizeof(T), std::ios_base::cur);
188  break;
189  case read:
190  F.read((char*) p, count * sizeof(T));
191  break;
192  }
193 
194  nrn_assert(!F.fail());
195  return p;
196  }

232  /** Write a padded array. nb_elements is number of elements to write per line,
233  * line_width is full size of a line in nb elements**/
234  template <typename T>
235  void write_array(T* p,
236  size_t nb_elements,
237  size_t line_width,
238  size_t nb_lines,
239  bool to_transpose = false) {
240  nrn_assert(F.is_open());
241  nrn_assert(current_mode & std::ios::out);
242  write_checkpoint();
243  T* temp_cpy = new T[nb_elements * nb_lines];
244 
245  if (to_transpose) {
246  for (size_t i = 0; i < nb_lines; i++) {
247  for (size_t j = 0; j < nb_elements; j++) {
248  temp_cpy[i + j * nb_lines] = p[i * line_width + j];
249  }
250  }
251  } else {
252  memcpy(temp_cpy, p, nb_elements * sizeof(T) * nb_lines);
253  }
254  // AoS never use padding, SoA is translated above, so one write
255  // operation is enought in both cases
256  F.write((const char*) temp_cpy, nb_elements * sizeof(T) * nb_lines);
257  nrn_assert(!F.fail());
258  delete[] temp_cpy;
259  }
