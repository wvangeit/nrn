111  int pc = (iwarp == 0); // print warp 0
112  pc = 0; // turn off printing
113  int nodebegin = ii.lastnode[iwarp];
114  int* stride = ii.stride + ii.stridedispl[iwarp];
115  int ncycle = ii.cellsize[iwarp];
116 
117  int inode = nodebegin;
118 
119  size_t nn = 0; // number of nodes in warp. '.'
120  size_t nx = 0; // number of idle cores on all cycles. 'X'
121  size_t ncacheline = 0;
122  ; // number of parent memory cacheline accesses.
123  // assmue warpsize is max number in a cachline so all o
124  size_t ncr = 0; // number of child race. nchild-1 of same parent in same cycle
125 
126  for (int icycle = 0; icycle < ncycle; ++icycle) {
127  int s = stride[icycle];
128  int lastp = -2;
129  if (pc)
130  printf(" ");
131  std::set<int> crace; // how many children have same parent in a cycle
132  for (int icore = 0; icore < warpsize; ++icore) {
133  char ch = '.';
134  if (icore < s) {
135  int par = p[inode];
136  if (crace.find(par) != crace.end()) {
137  ch = 'r';

139  } else {
140  crace.insert(par);
141  }
142 
143  if (par != lastp + 1) {
144  ch = (ch == 'r') ? 'R' : 'o';
145  ++ncacheline;
146  }
147  lastp = p[inode++];
148  ++nn;
149  } else {
150  ch = 'X';
151  ++nx;
152  }
153  if (pc)
154  printf("%c", ch);
155  }
156  if (pc)
157  printf("\n");
158  }
159 
160  ii.nnode[iwarp] = nn;
161  ii.ncycle[iwarp] = size_t(ncycle);
162  ii.idle[iwarp] = nx;
163  ii.cache_access[iwarp] = ncacheline;
164  ii.child_race[iwarp] = ncr;
165  if (pc)
166  printf("warp %d: %ld nodes, %d cycles, %ld idle, %ld cache access, %ld child races\n",
167  iwarp,
168  nn,
169  ncycle,

175 static void print_quality1(int iwarp, InterleaveInfo& ii, int ncell, int* p) {
176  int pc = ((iwarp == 0) || iwarp == (ii.nwarp - 1)); // warp not to skip printing
177  pc = 0; // turn off printing.
178  int* stride = ii.stride;
179  int cellbegin = iwarp * warpsize;
180  int cellend = cellbegin + warpsize;
181  cellend = (cellend < stride[0]) ? cellend : stride[0];
182 
183  int ncycle = 0;
184  for (int i = cellbegin; i < cellend; ++i) {
185  if (ncycle < ii.cellsize[i]) {
186  ncycle = ii.cellsize[i];
187  }
188  }
189  nrn_assert(ncycle == ii.cellsize[cellend - 1]);
190  nrn_assert(ncycle <= ii.nstride);
191 
192  int ncell_in_warp = cellend - cellbegin;
193 
194  size_t n = 0; // number of nodes in warp (not including roots)
195  size_t nx = 0; // number of idle cores on all cycles. X
196  size_t ncacheline = 0; // number of parent memory cacheline accesses. assume warpsize is max
197  // number in a cacheline so first core has all o
198 
199  int inode = ii.firstnode[cellbegin];
200  for (int icycle = 0; icycle < ncycle; ++icycle) {
201  int sbegin = ncell - stride[icycle] - cellbegin;
202  int lastp = -2;
203  if (pc)
204  printf(" ");
205  for (int icore = 0; icore < warpsize; ++icore) {
206  char ch = '.';
207  if (icore < ncell_in_warp && icore >= sbegin) {
208  int par = p[inode + icore];
209  if (par != lastp + 1) {
210  ch = 'o';
211  ++ncacheline;
212  }
213  lastp = par;
214  ++n;
215  } else {
216  ch = 'X';
217  ++nx;
218  }
219  if (pc)
220  printf("%c", ch);
221  }
222  if (pc)
223  printf("\n");
224  inode += ii.stride[icycle + 1];
225  }
226 
227  ii.nnode[iwarp] = n;
228  ii.ncycle[iwarp] = (size_t) ncycle;
229  ii.idle[iwarp] = nx;
230  ii.cache_access[iwarp] = ncacheline;
231  ii.child_race[iwarp] = 0;
232  if (pc)
233  printf("warp %d: %ld nodes, %d cycles, %ld idle, %ld cache access\n",
234  iwarp,
235  n,
236  ncycle,
237  nx,
238  ncacheline);
239 }

241 static void warp_balance(int ith, InterleaveInfo& ii) {
242  size_t nwarp = size_t(ii.nwarp);
243  size_t smm[4][3]; // sum_min_max see cp below
244  for (size_t j = 0; j < 4; ++j) {
245  smm[j][0] = 0;
246  smm[j][1] = 1000000000;
247  smm[j][2] = 0;
248  }
249  double emax = 0.0, emin = 1.0;
250  for (size_t i = 0; i < nwarp; ++i) {
251  size_t n = ii.nnode[i];
252  double e = double(n) / (n + ii.idle[i]);
253  if (emax < e) {
254  emax = e;
255  }
256  if (emin > e) {
257  emin = e;
258  }
259  size_t s[4] = {n, ii.idle[i], ii.cache_access[i], ii.child_race[i]};
260  for (size_t j = 0; j < 4; ++j) {
261  smm[j][0] += s[j];
262  if (smm[j][1] > s[j]) {
263  smm[j][1] = s[j];
264  }
265  if (smm[j][2] < s[j]) {
266  smm[j][2] = s[j];
267  }
268  }
269  }
270  std::vector<size_t> v(nwarp);
271  for (size_t i = 0; i < nwarp; ++i) {
272  v[i] = ii.ncycle[i];
273  }
274  double bal = load_balance(v);
275 #ifdef DEBUG
276  printf(
277  "thread %d nwarp=%ld balance=%g warp_efficiency %g to %g\n", ith, nwarp, bal, emin, emax);
278  const char* cp[4] = {"nodes", "idle", "ca", "cr"};
279  for (size_t i = 0; i < 4; ++i) {
280  printf(" %s=%ld (%ld:%ld)", cp[i], smm[i][0], smm[i][1], smm[i][2]);
281  }
282  printf("\n");
283 #else
284  (void) bal; // Remove warning about unused
285 #endif
286 }

288 int* interleave_order(int ith, int ncell, int nnode, int* parent) {
289  // return if there are no nodes to permute
290  if (nnode <= 0)
291  return nullptr;
292 
293  // ensure parent of root = -1
294  for (int i = 0; i < ncell; ++i) {
295  if (parent[i] == 0) {
296  parent[i] = -1;
297  }
298  }
299 
300  int nwarp = 0, nstride = 0, *stride = nullptr, *firstnode = nullptr;
301  int *lastnode = nullptr, *cellsize = nullptr, *stridedispl = nullptr;
302 
303  int* order = node_order(
304  ncell, nnode, parent, nwarp, nstride, stride, firstnode, lastnode, cellsize, stridedispl);
305 
306  if (interleave_info) {
307  InterleaveInfo& ii = interleave_info[ith];
308  ii.nwarp = nwarp;
309  ii.nstride = nstride;
310  ii.stridedispl = stridedispl;
311  ii.stride = stride;
312  ii.firstnode = firstnode;
313  ii.lastnode = lastnode;
314  ii.cellsize = cellsize;
315  if (0 && ith == 0 && interleave_permute_type == 1) {
316  printf("ith=%d nstride=%d ncell=%d nnode=%d\n", ith, nstride, ncell, nnode);
317  for (int i = 0; i < ncell; ++i) {
318  printf("icell=%d cellsize=%d first=%d last=%d\n",
319  i,
320  cellsize[i],
321  firstnode[i],
322  lastnode[i]);
323  }
324  for (int i = 0; i < nstride; ++i) {
325  printf("istride=%d stride=%d\n", i, stride[i]);
326  }
327  }
328  if (ith == 0) {
329  // needed for print_quality[12] and done once here to save time
330  int* p = new int[nnode];
331  for (int i = 0; i < nnode; ++i) {
332  p[i] = parent[i];
333  }
334  permute_ptr(p, nnode, order);
335  node_permute(p, nnode, order);
336 
337  ii.nnode = new size_t[nwarp];
338  ii.ncycle = new size_t[nwarp];
339  ii.idle = new size_t[nwarp];
340  ii.cache_access = new size_t[nwarp];
341  ii.child_race = new size_t[nwarp];
342  for (int i = 0; i < nwarp; ++i) {
343  if (interleave_permute_type == 1) {
344  print_quality1(i, interleave_info[ith], ncell, p);
345  }
346  if (interleave_permute_type == 2) {
347  print_quality2(i, interleave_info[ith], p);
348  }
349  }
350  delete[] p;
351  warp_balance(ith, interleave_info[ith]);
352  }
353  }
354 
355  return order;
356 }

358 #if INTERLEAVE_DEBUG // only the cell per core style
359 static int** cell_indices_debug(NrnThread& nt, InterleaveInfo& ii) {
360  int ncell = nt.ncell;
361  int nnode = nt.end;
362  int* parents = nt._v_parent_index;
363 
364  // we expect the nodes to be interleave ordered with smallest cell first
365  // establish consistency with ii.
366  // first ncell parents are -1
367  for (int i = 0; i < ncell; ++i) {
368  nrn_assert(parents[i] == -1);
369  }
370  int* sz = new int[ncell];
371  int* cell = new int[nnode];
372  for (int i = 0; i < ncell; ++i) {
373  sz[i] = 0;
374  cell[i] = i;
375  }
376  for (int i = ncell; i < nnode; ++i) {
377  cell[i] = cell[parents[i]];
378  sz[cell[i]] += 1;
379  }
380 
381  // cells are in inceasing sz order;
382  for (int i = 1; i < ncell; ++i) {
383  nrn_assert(sz[i - 1] <= sz[i]);
384  }
385  // same as ii.cellsize
386  for (int i = 0; i < ncell; ++i) {
387  nrn_assert(sz[i] == ii.cellsize[i]);
388  }
389 
390  int** cellindices = new int*[ncell];
391  for (int i = 0; i < ncell; ++i) {
392  cellindices[i] = new int[sz[i]];
393  sz[i] = 0; // restart sz counts
394  }
395  for (int i = ncell; i < nnode; ++i) {
396  cellindices[cell[i]][sz[cell[i]]] = i;
397  sz[cell[i]] += 1;
398  }
399  // cellindices first and last same as ii first and last
400  for (int i = 0; i < ncell; ++i) {
401  nrn_assert(cellindices[i][0] == ii.firstnode[i]);
402  nrn_assert(cellindices[i][sz[i] - 1] == ii.lastnode[i]);
403  }
404 
405  delete[] sz;
406  delete[] cell;
407 
408  return cellindices;
409 }

435 // each cell in parallel regardless of inhomogeneous topology
436 static void triang_interleaved(NrnThread* nt,
437  int icell,
438  int icellsize,
439  int nstride,
440  int* stride,
441  int* lastnode) {
442  int i = lastnode[icell];
443  for (int istride = nstride - 1; istride >= 0; --istride) {
444  if (istride < icellsize) { // only first icellsize strides matter
445  // what is the index
446  int ip = GPU_PARENT(i);
447 #ifndef CORENEURON_ENABLE_GPU
448  nrn_assert(ip >= 0); // if (ip < 0) return;
449 #endif
450  double p = GPU_A(i) / GPU_D(i);
451  GPU_D(ip) -= p * GPU_B(i);
452  GPU_RHS(ip) -= p * GPU_RHS(i);
453  i -= stride[istride];
454  }
455  }
456 }

459 static void bksub_interleaved(NrnThread* nt,
460  int icell,
461  int icellsize,
462  int /* nstride */,
463  int* stride,
464  int* firstnode) {

471 #endif

473  GPU_RHS(i) /= GPU_D(i);
474  i += stride[istride + 1];
475  }

478 nrn_pragma_acc(routine vector)
479 static void solve_interleaved2_loop_body(NrnThread* nt,
480  int icore,
481  int* ncycles,
482  int* strides,
483  int* stridedispl,
484  int* rootbegin,
485  int* nodebegin) {
486  int iwarp = icore / warpsize; // figure out the >> value
487  int ic = icore & (warpsize - 1); // figure out the & mask
488  int ncycle = ncycles[iwarp];
489  int* stride = strides + stridedispl[iwarp];
490  int root = rootbegin[iwarp]; // cell ID -> [0, ncell)
491  int lastroot = rootbegin[iwarp + 1];
492  int firstnode = nodebegin[iwarp];
493  int lastnode = nodebegin[iwarp + 1];
494 
495  // triang_interleaved2
496  {
497  int icycle = ncycle - 1;
498  int istride = stride[icycle];
499  int ii = lastnode - istride + ic;
500 
501  // execute until all tree depths are executed
502  bool has_subtrees_to_compute = true;
503 
504  nrn_pragma_acc(loop seq)
505  for (; has_subtrees_to_compute;) { // ncycle loop
506  // serial test, gpu does this in parallel
507  nrn_pragma_acc(loop vector)
508  nrn_pragma_omp(loop bind(parallel))
509  for (int icore = 0; icore < warpsize; ++icore) {
510  int i = ii + icore;
511  if (icore < istride) { // most efficient if istride equal warpsize
512  // what is the index
513  int ip = GPU_PARENT(i);
514  double p = GPU_A(i) / GPU_D(i);
515  nrn_pragma_acc(atomic update)
516  nrn_pragma_omp(atomic update)
517  GPU_D(ip) -= p * GPU_B(i);
518  nrn_pragma_acc(atomic update)
519  nrn_pragma_omp(atomic update)
520  GPU_RHS(ip) -= p * GPU_RHS(i);
521  }
522  }
523  // if finished with all tree depths then ready to break
524  // (note that break is not allowed in OpenACC)
525  if (icycle == 0) {
526  has_subtrees_to_compute = false;
527  continue;
528  }
529  --icycle;
530  istride = stride[icycle];
531  ii -= istride;
532  }
533  }
534  // bksub_interleaved2
535  auto const bksub_root = root + ic;
536  {
537  nrn_pragma_acc(loop seq)
538  for (int i = bksub_root; i < lastroot; i += 1) {
539  GPU_RHS(i) /= GPU_D(i); // the root
540  }
541 
542  int i = firstnode + ic;
543  int ii = i;
544  nrn_pragma_acc(loop seq)
545  for (int icycle = 0; icycle < ncycle; ++icycle) {
546  int istride = stride[icycle];
547  // serial test, gpu does this in parallel
548  nrn_pragma_acc(loop vector)
549  nrn_pragma_omp(loop bind(parallel))
550  for (int icore = 0; icore < warpsize; ++icore) {
551  int i = ii + icore;
552  if (icore < istride) {
553  int ip = GPU_PARENT(i);
554  GPU_RHS(i) -= GPU_B(i) * GPU_RHS(ip);
555  GPU_RHS(i) /= GPU_D(i);
556  }
557  i += istride;
558  }
559  ii += istride;
560  }
561  }
562 }

564 /**
565  * \brief Solve Hines matrices/cells with compartment-based granularity.
566  *
567  * The node ordering/permuation guarantees cell interleaving (as much coalesced memory access as
568  * possible) and balanced warps (through the use of lpt algorithm to define the groups/warps). Every
569  * warp deals with a group of cells, therefore multiple compartments (finer level of parallelism).
570  */
571 void solve_interleaved2(int ith) {
572  NrnThread* nt = nrn_threads + ith;
573  InterleaveInfo& ii = interleave_info[ith];
574  int nwarp = ii.nwarp;
575  if (nwarp == 0)
576  return;
577 
578  int ncore = nwarp * warpsize;
579 
580 #ifdef _OPENACC
581  if (corenrn_param.gpu && corenrn_param.cuda_interface) {
582  auto* d_nt = static_cast<NrnThread*>(acc_deviceptr(nt));
583  auto* d_info = static_cast<InterleaveInfo*>(acc_deviceptr(interleave_info + ith));
584  solve_interleaved2_launcher(d_nt, d_info, ncore, acc_get_cuda_stream(nt->stream_id));
585  } else {
586 #endif
587  int* ncycles = ii.cellsize; // nwarp of these
588  int* stridedispl = ii.stridedispl; // nwarp+1 of these
589  int* strides = ii.stride; // sum ncycles of these (bad since ncompart/warpsize)
590  int* rootbegin = ii.firstnode; // nwarp+1 of these
591  int* nodebegin = ii.lastnode; // nwarp+1 of these
592 #if defined(CORENEURON_ENABLE_GPU)
593  int nstride = stridedispl[nwarp];
594 #endif
595  // nvc++/22.3 does not respect an if clause inside nrn_pragma_omp...
596  if (nt->compute_gpu) {
597  /* If we compare this loop with the one from cellorder.cu (CUDA version), we will
598  * understand that the parallelism here is exposed in steps, while in the CUDA version
599  * all the parallelism is exposed from the very beginning of the loop. In more details,
600  * here we initially distribute the outermost loop, e.g. in the CUDA blocks, and for the
601  * innermost loops we explicitly use multiple threads for the parallelization (see for
602  * example the loop directives in triang/bksub_interleaved2). On the other hand, in the
603  * CUDA version the outermost loop is distributed to all the available threads, and
604  * therefore there is no need to have the innermost loops. Here, the loop/icore jumps
605  * every warpsize, while in the CUDA version the icore increases by one. Other than
606  * this, the two loop versions are equivalent (same results).
607  */
608  nrn_pragma_acc(parallel loop gang present(nt [0:1],
609  strides [0:nstride],
610  ncycles [0:nwarp],
611  stridedispl [0:nwarp + 1],
612  rootbegin [0:nwarp + 1],
613  nodebegin [0:nwarp + 1]) async(nt->stream_id))
614  // clang-format off
615  nrn_pragma_omp(target teams loop map(present, alloc: nt[:1],
616  strides[:nstride],
617  ncycles[:nwarp],
618  stridedispl[:nwarp + 1],
619  rootbegin[:nwarp + 1],
