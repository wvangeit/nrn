227 void nrnmpi_subworld_size(int n) {
228  /* n is the (desired) size of a subworld (pc.nhost) */
229  /* A subworld (net) is contiguous */
230  /* In case pc.nhost_world/n is not an integer, there are
231  pc.nhost_world/n + 1 subworlds and the last subworld
232  has pc.nhost_world%n ranks. All the other subworlds have n ranks.
233  */
234  if (nrnmpi_use != 1) {
235  return;
236  }
237  if (nrnmpi_comm != MPI_COMM_NULL) {
238  nrn_mpi_assert(MPI_Comm_free(&nrnmpi_comm));
239  nrnmpi_comm = MPI_COMM_NULL;
240  }
241  if (nrn_bbs_comm != MPI_COMM_NULL) {
242  nrn_mpi_assert(MPI_Comm_free(&nrn_bbs_comm));
243  nrn_bbs_comm = MPI_COMM_NULL;
244  }
245  if (grp_bbs != MPI_GROUP_NULL) {
246  nrn_mpi_assert(MPI_Group_free(&grp_bbs));
247  grp_bbs = MPI_GROUP_NULL;
248  }
249  if (grp_net != MPI_GROUP_NULL) {
250  nrn_mpi_assert(MPI_Group_free(&grp_net));
251  grp_net = MPI_GROUP_NULL;
252  }
253  MPI_Group wg;
254  nrn_mpi_assert(MPI_Comm_group(nrnmpi_world_comm, &wg));
255  int r = nrnmpi_myid_world;
256  /* special cases */
257  if (n == 1) {
258  nrn_mpi_assert(MPI_Group_incl(wg, 1, &r, &grp_net));
259  nrn_mpi_assert(MPI_Comm_dup(nrnmpi_world_comm, &nrn_bbs_comm));
260  nrn_mpi_assert(MPI_Comm_create(nrnmpi_world_comm, grp_net, &nrnmpi_comm));
261  nrn_mpi_assert(MPI_Comm_rank(nrnmpi_comm, &nrnmpi_myid));
262  nrn_mpi_assert(MPI_Comm_size(nrnmpi_comm, &nrnmpi_numprocs));
263  nrn_mpi_assert(MPI_Comm_rank(nrn_bbs_comm, &nrnmpi_myid_bbs));
264  nrn_mpi_assert(MPI_Comm_size(nrn_bbs_comm, &nrnmpi_numprocs_bbs));
265  nrnmpi_subworld_id = nrnmpi_myid_bbs;
266  nrnmpi_numprocs_subworld = nrnmpi_numprocs_bbs;
267  } else if (n == nrnmpi_numprocs_world) {
268  nrn_mpi_assert(MPI_Group_incl(wg, 1, &r, &grp_bbs));
269  nrn_mpi_assert(MPI_Comm_dup(nrnmpi_world_comm, &nrnmpi_comm));
270  nrn_mpi_assert(MPI_Comm_create(nrnmpi_world_comm, grp_bbs, &nrn_bbs_comm));
271  nrn_mpi_assert(MPI_Comm_rank(nrnmpi_comm, &nrnmpi_myid));
272  nrn_mpi_assert(MPI_Comm_size(nrnmpi_comm, &nrnmpi_numprocs));
273  if (r == 0) {
274  nrn_mpi_assert(MPI_Comm_rank(nrn_bbs_comm, &nrnmpi_myid_bbs));
275  nrn_mpi_assert(MPI_Comm_size(nrn_bbs_comm, &nrnmpi_numprocs_bbs));
276  } else {
277  nrnmpi_myid_bbs = -1;
278  nrnmpi_numprocs_bbs = -1;
279  }
280  nrnmpi_subworld_id = 0;
281  nrnmpi_numprocs_subworld = nrnmpi_numprocs;
282  } else {
283  int nw = nrnmpi_numprocs_world;
284  int nb = nw / n; /* nrnmpi_numprocs_bbs */
285  int ib;
286  int range[3];
287  if (nw % n) {
288  nb += 1; /* and the last will have pc.nhost = nw%n */
289  }
290  /* A subworld (net) has contiguous ranks. */
291  /* Every rank is in a specific nrnmpi_comm communicator */
292  ib = r / n;
293  range[0] = ib * n; /* first rank in group */
294  range[1] = range[0] + n - 1; /* last rank in group */
295  if (range[1] >= nw) {
296  range[1] = nw - 1;
297  }
298  range[2] = 1; /* stride */
299  nrn_mpi_assert(MPI_Group_range_incl(wg, 1, &range, &grp_net));
300  nrn_mpi_assert(MPI_Comm_create(nrnmpi_world_comm, grp_net, &nrnmpi_comm));
301  nrn_mpi_assert(MPI_Comm_rank(nrnmpi_comm, &nrnmpi_myid));
302  nrn_mpi_assert(MPI_Comm_size(nrnmpi_comm, &nrnmpi_numprocs));
303 
304  /* nrn_bbs_com ranks stride is nrnmpi_numprocs */
305  /* only rank 0 of each subworld participates in nrn_bbs_comm */
306  range[0] = 0; /* first world rank in nrn_bbs_comm */
307  range[1] = (nb - 1) * n; /* last world rank in nrn_bbs_comm */
308  range[2] = n; /* stride */
309  nrn_mpi_assert(MPI_Group_range_incl(wg, 1, &range, &grp_bbs));
310  nrn_mpi_assert(MPI_Comm_create(nrnmpi_world_comm, grp_bbs, &nrn_bbs_comm));
311  if (r % n == 0) { /* only rank 0 participates in nrn_bbs_comm */
312  nrn_mpi_assert(MPI_Comm_rank(nrn_bbs_comm, &nrnmpi_myid_bbs));
313  nrn_mpi_assert(MPI_Comm_size(nrn_bbs_comm, &nrnmpi_numprocs_bbs));
314  } else {
315  nrnmpi_myid_bbs = -1;
316  nrnmpi_numprocs_bbs = -1;
317  }
318  nrnmpi_subworld_id = r / n;
319  nrnmpi_numprocs_subworld = n;
320  if ((nw % n != 0) && (nrnmpi_subworld_id == (nb - 1))) {
321  nrnmpi_numprocs_subworld = nw % n; /* and the last will have pc.nhost = nw%n */
322  }
323  }
324  nrnmpi_subworld_change_cnt++;
325  nrn_mpi_assert(MPI_Group_free(&wg));
326 }
