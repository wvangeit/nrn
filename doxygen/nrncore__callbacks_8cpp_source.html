886 static void set_info(TQItem* tqi,
887  int tid,
888  NrnCoreTransferEvents* core_te,
889  std::unordered_map<NetCon*, std::vector<size_t>>& netcon2intdata,
890  std::unordered_map<PreSyn*, std::vector<size_t>>& presyn2intdata,
891  std::unordered_map<double*, std::vector<size_t>>& weight2intdata) {
892  DiscreteEvent* de = (DiscreteEvent*) (tqi->data_);
893  int type = de->type();
894  double tdeliver = tqi->t_;
895  core_te->type.push_back(type);
896  core_te->td.push_back(tdeliver);
897 
898  switch (type) {
899  case DiscreteEventType: { // 0
900  } break;
901  case NetConType: { // 2
902  NetCon* nc = (NetCon*) de;
903  // To find the i for cg.netcons[i] == nc
904  // and since there are generally very many fewer nc on the queue
905  // than netcons. Begin a nc2i map that we can fill in for i
906  // later in one sweep through the cg.netcons.
907  // Here is where it goes.
908  size_t iloc = core_te->intdata.size();
909  core_te->intdata.push_back(-1);
910  // But must take into account the rare situation where the same
911  // NetCon is on the queue more than once. Hence the std::vector
912  netcon2intdata[nc].push_back(iloc);
913  } break;
914  case SelfEventType: { // 3
915  SelfEvent* se = (SelfEvent*) de;
916  Point_process* pnt = se->target_;
917  int type = pnt->prop->_type;
918  int movable_index = type2movable[type];
919  double* wt = se->weight_;
920 
921  core_te->intdata.push_back(type);
922  core_te->dbldata.push_back(se->flag_);

925  // it was issued in response to a NetCon event to the target
926  // NET_RECEIVE block. Determination of Point_process* target_ on the
927  // CoreNEURON side uses mechanism type and instance index from here
928  // on the NEURON side. And the latter can be determined now from pnt.
929  // On the other hand, if there is a non-null weight pointer, its index
930  // can only be determined by sweeping over all NetCon.
931 
932  // Introduced the public static method below because ARTIFICIAL_CELL
933  // are not located in NrnThread and are not cache efficient.
934  int index = CellGroup::nrncore_pntindex_for_queue(pnt->prop, tid, type);
935  core_te->intdata.push_back(index);
936 
937  size_t iloc_wt = core_te->intdata.size();
938  if (wt) { // don't bother with NULL weights
939  weight2intdata[wt].push_back(iloc_wt);
940  }
941  core_te->intdata.push_back(-1); // If NULL weight this is the indicator
942  // Each of these holds a TQItem*
943  Datum* const movable = se->movable_;
944  Datum* const pnt_movable = pnt->prop->dparam + movable_index;

946  // movable

949  if (condition) {

953  } break;

958  // Skip if PreSyn not associated with this thread.
959  bool skip = (ps->nt_ && (ps->nt_->id != tid)) ? true : false;
960  // Skip if effectively an InputPresyn (ps->nt_ == NULL)
961  // and this is not thread 0.
962  skip = (!ps->nt_ && tid != 0) ? true : skip;
963  if (skip) {

965  core_te->type.pop_back();
966  core_te->td.pop_back();

970  // Input PreSyn (ps->output_index = -1 and ps->gid >= 0)
971  // is distinquished from PreSyn (ps->output_index == ps->gid
972  // or both negative) by the first item of 0 or 1 respectively followed
973  // by gid or presyn index respectively.
974  // That is:
975  // Output PreSyn format is 0, presyn index

981  core_te->intdata.push_back(1);
982  core_te->intdata.push_back(ps->gid_);

991  // is in effect, then the send time is actually tt - nc->delay_
992  // (Note there is no core2nrn inverse as PreSyn does not appear on

994  if (ps->use_min_delay_) {

1081  }
1082  presyn2intdata[ps][0] = i + NRN_SENTINAL;
1083  }
1084  }

1086  // presyn2intdata[ps][0] must be >= NRN_SENTINAL
